    $q: String!
    $nestedQueryList: [SPARK_NestedQueryInput]
    $page: Int!
    $id: ID!
    $sortField: String
    $sortDirection: SPARK_SortDirection
    $options: SPARK_MaterialsSearchOptions
    $isInternal: Boolean = false
  ) {
    searchMaterialsNested(q: $q, nestedQueryList: $nestedQueryList, page: $page, size: 48, sortField: $sortField, sortDirection: $sortDirection, options: $options) {
      ...MaterialsSearchDetails
    }
    getUserAccessUnencoded(resourceIds: [$id]) {
      resourcePermissions {
        permissions
      }
    }
    getUserSettings {
      materialListView
    }
