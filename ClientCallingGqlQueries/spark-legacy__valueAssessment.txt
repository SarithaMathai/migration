import { gql } from '@apollo/client'

const FRAGMENT_VALUE_ASSESSMENT_ERRORS = gql`
  fragment fragmentValueAssessmentErrors on SPARK_ValueAssessmentErrors {
    code
    field
    reason
    resource
  }
`

const FRAGMENT_VALUE_ASSESSMENT_RECORD = gql`
  fragment fragmentValueAssessmentRecord on SPARK_ValueAssessmentRecord {
    id
    name
    isFavorite
    isOriginal
    consumptionUnitType
    columns {
      field
      headerName
    }
    rows {
      id
      uuid
      type
      isResolved
      costGroup
      active
      costs {
        columnFieldId
        cost
        consumption
      }
    }
  }
`

const FRAGMENT_VALUE_ASSESSMENT = gql`
  fragment fragmentValueAssessment on SPARK_ValueAssessment {
    id
    createdAt
    updatedAt
    createdBy
    plannedRetailPrice
    originalAssessment {
      ...fragmentValueAssessmentRecord
    }
    scenarioAssessments {
      ...fragmentValueAssessmentRecord
    }
  }
  ${FRAGMENT_VALUE_ASSESSMENT_RECORD}
`

export const GET_SAMPLE_VALUE_ASSESSMENT = gql`
  query getSampleValueAssessment($sampleId: String!) {
    getSampleValueAssessment(sampleId: $sampleId) {
      message
      errors {
        ...fragmentValueAssessmentErrors
      }
      data {
        ...fragmentValueAssessment
      }
    }
  }
  ${FRAGMENT_VALUE_ASSESSMENT_ERRORS}
  ${FRAGMENT_VALUE_ASSESSMENT}
`

export const SEARCH_VALUE_ASSESSMENT_MATERIALS = gql`
  query searchValueAssessmentMaterials($valueAssessmentId: String!, $q: String) {
    searchValueAssessmentMaterials(valueAssessmentId: $valueAssessmentId, q: $q) {
      message
      totalItems
      data {
        id
        description
        humanId
      }
      errors {
        ...fragmentValueAssessmentErrors
      }
    }
  }
  ${FRAGMENT_VALUE_ASSESSMENT_ERRORS}
`

export const CREATE_VALUE_ASSESSMENT_MOCK_FABRIC = gql`
  mutation createValueAssessmentMockFabric(
    $valueAssessmentId: String!
    $requirementId: String!
    $materialCategoryId: String!
    $fabricWidthUnit: String!
    $description: String!
    $cost: Float!
    $fabricWidth: Float!
  ) {
    createValueAssessmentMockFabric(
      valueAssessmentId: $valueAssessmentId
      requirementId: $requirementId
      materialCategoryId: $materialCategoryId
      fabricWidthUnit: $fabricWidthUnit
      description: $description
      cost: $cost
      fabricWidth: $fabricWidth
    ) {
      data {
        id
        requirementId
        description
      }
      message
      errors {
        ...fragmentValueAssessmentErrors
      }
    }
  }
  ${FRAGMENT_VALUE_ASSESSMENT_ERRORS}
`
