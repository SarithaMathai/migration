    searchAttachments(q: $q, parentIds: $parentIds, relatedIds: $relatedIds, partnerId: $partnerId, page: $page, size: $size) {
      paging {
        size
        totalPages
        totalElements
        number
      }
      content {
        id
        documentId
        humanId
        documentType
        name
        resource {
          id
          type
        }
        parentResource
        relatedProduct {
          id
          description
        }
        relatedWorkspace {
          id
          description
        }
        relatedResources
        previewable
      }
    }
  }
`
