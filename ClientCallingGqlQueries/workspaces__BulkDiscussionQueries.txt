import { gql } from '@apollo/client'

export const GET_BULK_DISCUSSION_DATA = gql`
  query getBulkDiscussionData($productIds: [ID]!, $workspaceId: String) {
    getRoles {
      roles {
        id
        value
      }
    }
    getProductsByIds(ids: $productIds) {
      paging {
        totalElements
        totalPages
      }
      content {
        id
        description
        thumbnailId
        teams(workspaceId: $workspaceId) {
          content {
            id
            partnerId: partner_id
            partnerName: partner_name
            partnerType {
              code
            }
            name
            members {
              roles
              memberDetails {
                id
                active
                companyName
                firstName
                lastName
              }
            }
          }
        }
        businessPartners {
          id
          name
          associatedPartnerType
        }
        droppedPartnerIds
      }
    }
  }
`
