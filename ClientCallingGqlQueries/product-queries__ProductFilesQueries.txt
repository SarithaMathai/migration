export const GET_PRODUCT_WITH_ATTACHMENTS_AND_COMPONENTS = gql`
  query getProductWithAttachmentsAndComponents(
    $id: ID!
    $partnerId: ID
    $type: String
    $tags: [String]
    $archived: Boolean = false
    $workspaceId: String
    $includeUnfilteredData: Boolean = false
    $onlyProductPacketFiles: Boolean = false
    $types: [Int] = [100]
  ) {
    getProduct(id: $id) {
      id
      ...ProductBaseInfoFragment
      claims {
        humanId
        workspaceContext
        claimName
        claimDetails {
          id
          vendorSubstantiate {
            substantiated
            substantiatedAt
            substantiatedBy {
              id
              firstName
              lastName
              internal: isInternal
            }
          }
        }
        statuses {
          workspaceId
          type
          overallStatus {
            code
            description
          }
          status {
            code
            description
          }
        }
      }
      attachmentsV3(partnerId: $partnerId, type: $type, tags: $tags, onlyProductPacketFiles: $onlyProductPacketFiles) {
        counts {
          attachmentCount(skipQuery: true) {
            bpType
            counts
          }
        }
        content {
          ...AttachmentV3Fragment
        }
      }
      components(workspaceId: $workspaceId, partnerId: $partnerId, type: $type, tags: $tags, archived: $archived, types: $types) {
        countByComponents {
          type
          counts
          countByStatus {
            code
            counts
            workspaceContext
          }
        }
        counts {
          bpType
          counts
        }
        archivedCount
        content {
          ...productComponentFragment
          rows {
            id
            templateId
          }
          currentUserClaimPermissions
        }
      }
      unfilteredAttachments: attachmentsV3(onlyProductPacketFiles: $onlyProductPacketFiles) @include(if: $includeUnfilteredData) {
        content {
          ...AttachmentV3Fragment
        }
      }
      unfilteredComponents: components @include(if: $includeUnfilteredData) {
        content {
          ...productComponentFragment
        }
      }
    }
  }
  ${PRODUCT_BASE_INFO_FRAGMENT}
  ${PRODUCT_COMPONENT_FRAGMENT}
  ${ATTACHMENT_V3_FRAGMENT}
`
