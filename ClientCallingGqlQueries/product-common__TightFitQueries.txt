export const GET_TIGHT_FITS = gql`
  ${TIGHT_FIT_FRAGMENT}
  query getTightFits($ids: [String], $name: String, $statusIds: [Int], $brandIds: [Int], $divisionIds: [Int], $departmentIds: [Int]) {

      tightFits {
        ...tightFitFragment

`

      humanId

  }
