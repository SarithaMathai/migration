export const MultiColorSampleEditBody = ({
  formatValues,
  sampleType,
  resetSampleType,
  isInternal,
  currentUser,
  sampleId,
  workspaceId,
  filterCriteria,
  productDetailsExpandable,
  onExpand,
  parentResourceType,
  parentHumanId,
  fabricSpecComboId,
  accreditations = [],
}) => {
  const { currentUserAccessToEvaluate, accreditationsWithStatus } = useAccreditation()

  const [loadCombination, { data: combination, called: loadCombinationCalled }] = useLazyQuery(GET_COMBINATION_BY_ID(isInternal), { variables: { id: parentHumanId } })

  const [loadFabricSpecCombo, { data: fabricSpecCombo, called: loadFabricSpecComboCalled }] = useLazyQuery(GET_FABRIC_SPEC_COMBO_BY_ID(isInternal), { variables: { id: fabricSpecComboId } })

  const location = useLocation()

  const { loading, error, data } = useQuery(GET_MCE_SAMPLE_WITH_TYPES, {
    variables: { ids: [sampleId] },
    fetchPolicy: 'network-only',
  })

  const [updateSample, { error: mutationError }] = useMutation(
    sampleType === `${SAMPLE_TYPES.MultiColorSample}` || sampleType === `${SAMPLE_TYPES.MaterialMultiColorSample}` || sampleType === `${SAMPLE_TYPES.Colors}` ? UPDATE_MCE_SAMPLE : UPDATE_SAMPLE,
    {
      awaitRefetchQueries: true,
      refetchQueries: [
        {
          query: GET_SAMPLE_OVERVIEW,
          variables: {
            ids: [sampleId],
          },
        },
        {
          query: GET_MCE_SAMPLE_WITH_TYPES,
          variables: {
            ids: [sampleId],
          },
        },
        ...(workspaceId
          ? [
              {
                query: GET_WORKSPACE_PRODUCTS_WITH_IDS,
                variables: {
                  page: 0,
                  size: WORKSPACE_PRODUCTS_LIST_PAGE_SIZE,
                  id: workspaceId,
                  q: '',
                  filter: filterCriteria,
                },
              },
            ]
          : []),
      ],
    },
  )

  useEffect(() => {
    if (parentResourceType === RESOURCE_TYPES.Combinations && !loadCombinationCalled) {
      loadCombination()
    }
  }, [parentResourceType, loadCombinationCalled, loadCombination])

  useEffect(() => {
    if (parentResourceType === RESOURCE_TYPES.Combinations && !loadFabricSpecComboCalled && fabricSpecComboId) {
      loadFabricSpecCombo()
    }
  }, [parentResourceType, loadFabricSpecComboCalled, fabricSpecComboId, loadFabricSpecCombo])

  if (loading) return <Loading>Loading multi-Color sample</Loading>
  if (error) return <Message>{ERROR_MESSAGE.MESSAGE_OTHER}</Message>
  if (mutationError) return <Message>{ERROR_MESSAGE.MESSAGE_OTHER}</Message>

  const sample = data && data.getSamplesByIdsV2 && data.getSamplesByIdsV2[0]

  const sampleEvalStatus = _.get(data, 'getProductSampleEvaluationTypesV2', [])
  const lateReason = _.get(data, 'getSampleLateReasonTypesV2', [])
  const trackingStatus = _.get(data, 'getSampleTrackingTypesV2', [])

  return (
    <Card className="hc8-pa-3x">
      <Grid.Container>
        <Grid.Item size={{ xs: 12 }}>
          <Heading size={4}>
            Editing {sample && sample.id} &mdash; {sample && sample.description}
          </Heading>
        </Grid.Item>
        <Grid.Item size={{ xs: 12 }}>
          <SampleEditForm
            partnerName={sample?.businessPartner?.name || ''}
            parentResourceType={parentResourceType}
            resetSampleType={resetSampleType}
            sampleTypeOptions={_.get(data, parentResourceType === RESOURCE_TYPES.Combinations ? 'getSampleMaterialTypesV2' : 'getSampleProductTypesV2', [])}
            trackingOptions={formatValues(trackingStatus, 'Status')}
            lateReasonOptions={lateReason}
            sampleFormats={_.get(data, parentResourceType === RESOURCE_TYPES.Product && 'getSampleFormats', [])}
            evaluationPurposes={_.get(data, parentResourceType === RESOURCE_TYPES.Product && 'getSampleEvaluationPurposes', [])}
            evaluationOptions={formatValues(sampleEvalStatus, 'Status')}
            updateSample={updateSample}
            isInternal={isInternal}
            currentUser={currentUser}
            sample={sample}
            combination={(combination && combination.getCombination) || {}}
            fabricSpecCombo={fabricSpecCombo}
            expandable={productDetailsExpandable}
            onExpand={onExpand}
            accreditationsWithStatus={accreditationsWithStatus}
            currentUserAccessToEvaluate={currentUserAccessToEvaluate}
            accreditations={accreditations}
            location={location}
          />
        </Grid.Item>
      </Grid.Container>
    </Card>
  )
}
