export const GET_USERS = gql`
  query getUsers($name: String, $excludeIds: [String], $partnerId: [ID]) {
    getUsers(page: 0, size: 7, name: $name, excludeIds: $excludeIds, partnerId: $partnerId) {
      content {
        userProfileId
        firstName
        lastName
        companyName
        internal: isInternal
        partnerId
        primaryRole
        currentVMMRole
        outOfOfficeAttributes {
          userOutOfOffice
        }
        accreditations {
          code
          description
          status
        }
      }
    }
  }
`
