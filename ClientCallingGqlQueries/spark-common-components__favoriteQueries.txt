import { gql } from '@apollo/client'

export const GET_FAVORITES_TEAMS = gql`
  query getFavoriteTeams($types: [String]!) {
    getFavoritesByTypes(types: $types, page: 0, size: 1000) {
      content {
        id
        resource {
          id
          type
        }
        team {
          name
        }
      }
    }
  }
`
