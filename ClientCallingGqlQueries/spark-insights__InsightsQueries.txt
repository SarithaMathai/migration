import { gql } from '@apollo/client'

export const GET_CARD_BY_ID = (fragment, queryName = 'getCard') => {
  const fragmentName = fragment.definitions[0].name.value
  return gql`
    query ${queryName}($id: String!, $filtersInput: Spark_Greenfield_Filters_Input) {
      getCard(id: $id, filtersInput: $filtersInput) {
        queryResults {
          ...${fragmentName}
        }
        lastUpdatedTimeStamp
      }
    }
    ${fragment}
  `
}
