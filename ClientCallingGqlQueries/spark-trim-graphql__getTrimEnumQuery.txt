import { TRIM_TYPES } from '@spark/trim-base'
import { gql } from '@apollo/client'

export const getTrimEnumQuery = (typeCode) => {
  switch (typeCode) {
    case TRIM_TYPES.BUCKLE:
      return GET_TRIM_BUCKLE_ENUMS
    case TRIM_TYPES.ELASTIC:
      return GET_TRIM_ELASTIC_ENUMS
    case TRIM_TYPES.ZIPPER:
      return GET_TRIM_ZIPPER_ENUMS
    case TRIM_TYPES.BUTTON:
      return GET_TRIM_BUTTON_ENUMS
    case TRIM_TYPES.CLOSURE:
      return GET_TRIM_CLOSURE_ENUMS
    case TRIM_TYPES.CONSTRUCTION_COMPONENT:
      return GET_TRIM_CONSTRUCTION_COMPONENTS_ENUMS
    case TRIM_TYPES.CORD_PARTS:
      return GET_TRIM_CORD_PARTS_ENUMS
    case TRIM_TYPES.EMBELLISHMENT:
      return GET_TRIM_EMBELLISHMENT_PARTS_ENUMS
    case TRIM_TYPES.HARDWARE:
      return GET_TRIM_HARDWARE_ENUMS
    case TRIM_TYPES.INTERLINING:
      return GET_TRIM_INTERLINING_ENUMS
    case TRIM_TYPES.LACE_NARROW:
      return GET_TRIM_LACE_NARROW_ENUMS
    case TRIM_TYPES.TAPE_RIBBON_CORDING:
      return GET_TRIM_TAPE_RIBBON_CORDING_ENUMS
    case TRIM_TYPES.THREAD:
      return GET_TRIM_THREAD_ENUMS
    case TRIM_TYPES.PATCH:
      return GET_TRIM_PATCH_ENUMS
    case TRIM_TYPES.ZIPPER_PULLER:
      return GET_TRIM_ZIPPER_PULLER_ENUMS
    case TRIM_TYPES.LABEL:
      return GET_TRIM_LABEL_ENUMS
  }
}

export const GET_TRIM_BUCKLE_ENUMS = gql`
query getBuckleEnums {
  getTrimSubTypesByType(typeCode: ${TRIM_TYPES.BUCKLE}) {
    code
    description
  }
  getFiberContentsByType(typeCode: ${TRIM_TYPES.BUCKLE}) {
    code
    description
  }
}
`
