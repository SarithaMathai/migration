import { gql } from '@apollo/client'

export const GET_CAPACITY_TYPES = gql`
  query getCapacityTypes($departmentId: ID!) {
    getCapacityTypes(departmentId: $departmentId) {
      capacityTypePrimaryId
      capacityTypePrimaryName
      capacityTypeSecondaryId
      capacityTypeSecondaryName
    }
  }
`

export const GET_BOM_COMPONENT_STATUS = gql`
  query getBomComponentStatus($ids: [String!]) {
    getBomByIds(ids: $ids) {
      humanId
      statuses {
        workspaceId
        type
        status {
          code
          description
        }
      }
    }
  }
`

export const GET_VERSIONED_COMPONENT_STATUS = gql`
  query getVersionedComponentStatus($id: String!, $version: Int!) {
    getComponentVersion(id: $id, version: $version) {
      ... on SPARK_Bom {
        humanId
        statuses {
          workspaceId
          status {
            code
            description
          }
        }
      }
      ... on SPARK_Measurements {
        humanId
        statuses {
          workspaceId
          status {
            code
            description
          }
        }
      }
      ... on SPARK_ProductDetails {
        humanId
        statuses {
          workspaceId
          status {
            code
            description
          }
        }
      }
    }
  }
`
