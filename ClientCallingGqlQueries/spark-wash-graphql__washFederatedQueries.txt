import { gql } from '@apollo/client'
import { TypedDocumentNode } from '@graphql-typed-document-node/core'
import { SPARK_CodeDescription } from 'Base'
import type { I_PLM_WASH, PLM_Fabric, PLM_FabricSpecification } from 'Wash'

// ── Query variable types ───────────────────────────────────────────────────────

export interface PlmGetWashQueryVariables {
  id: string
  includeFabricData?: boolean
  includeFabricSpecData?: boolean
}

export interface PlmGetWashStatusQueryVariables {
  id: string
}

export interface PlmGetWashFabricQueryVariables {
  id: string
}

export interface PlmExportWashQueryVariables {
  q: string
  nestedQueryList?: unknown[] | null
  isThumbnailsRequired?: boolean
}

// ── Mutation input types (mirror WashInput in plm-materials schema) ────────────

export interface PlmWashAdditionalInfoInput {
  designCycles?: string[]
  tags?: string[]
  notes?: string
  vendorWashCode?: string
}

export interface PlmWashDetailsInput {
  washType?: number | null
  sustainability?: number[]
  washProcess?: number[]
  garmentDye?: number[]
  colors?: string[]
  reasonForDeviation?: string
  totalLiquorRatio?: number
}

export interface PlmWashEimScoreInput {
  waterUse?: number | null
  chemicalUse?: number | null
  powerUse?: number | null
  laborUse?: number | null
  totalScore?: number | null
  isDevelopmentSample?: boolean
  isBulkProduction?: boolean
}

export interface PlmWashFabricDetailsInput {
  fabricId?: string | null
  fabricSpecificationIds?: string[]
}

export interface PlmTeamToAddInput {
  teamId?: string
  teamType?: string
  accessControlGroupIds?: string[]
  partnerId?: string
  partnerType?: number
}

/** Matches the federated `WashInput` accepted by `addWash` / `updateWash` on plmClient. */
export interface PlmWashInput {
  /** Required for updateWash; omit for addWash */
  id?: string | null
  washDescription?: string
  washNameOrColor?: string
  brand?: string
  departments?: string[]
  laundryFacilityId?: string | null
  isTargetRecipe?: boolean
  isYarnDrivenProduct?: boolean
  recipe?: number | null
  status?: number | string
  thumbnailId?: string | null
  designThumbnailId?: string | null
  silhouette?: string | null
  energyUsed?: number | null
  waterSavingPercent?: number | null
  coTwoSavingPercent?: number | null
  files?: string[]
  additionalInfo?: PlmWashAdditionalInfoInput
  washDetails?: PlmWashDetailsInput
  eimScore?: PlmWashEimScoreInput
  fabricDetails?: PlmWashFabricDetailsInput
  teamsToAdd?: PlmTeamToAddInput[] | string[]
}

export interface PlmAddWashMutationVariables {
  wash: PlmWashInput
}

export interface PlmUpdateWashMutationVariables {
  updateWash: PlmWashInput
}

export interface PlmTeamsToUpdateInput {
  teamsToAdd?: PlmTeamToAddInput[]
  teamsToRemove?: PlmTeamToAddInput[]
}

export interface PlmUpdateWashTeamsMutationVariables {
  id: string
  teamsToUpdate?: PlmTeamsToUpdateInput
}

// ── patchWash input types ──────────────────────────────────────────────────────

/**
 * Wrapper for ID fields that support explicit clearing.
 * - Set a value: { value: "ATT-ABC123" }
 * - Clear the field: { clear: true }
 * - Leave unchanged: omit the field from WashPatch entirely
 */
export interface PlmIDOrClear {
  value?: string | null
  clear?: boolean
}

/**
 * Partial update input for patchWash.
 * Only included fields are changed; omitted fields remain untouched.
 */
export interface PlmWashPatch {
  thumbnailId?: PlmIDOrClear
  designThumbnailId?: PlmIDOrClear
}

export interface PlmPatchWashMutationVariables {
  id: string
  patch: PlmWashPatch
}

export interface PlmPatchWashMutationData {
  patchWash: I_PLM_WASH
}

export interface PlmUpdateWashTeamsMutationData {
  updateWashTeams: boolean | null
}

// ── Mutation data types ────────────────────────────────────────────────────────

export interface PlmAddWashMutationData {
  addWash: I_PLM_WASH
}

export interface PlmUpdateWashMutationData {
  updateWash: I_PLM_WASH
}

// ── Query data types ───────────────────────────────────────────────────────────

export interface PlmGetWashQueryData {
  wash: I_PLM_WASH | null
}

export interface PlmGetWashStatusQueryData {
  wash: {
    id: string
    status: SPARK_CodeDescription
  } | null
}

export interface PlmGetWashFabricQueryData {
  wash: {
    id: string
    fabric?: PLM_Fabric | null
    fabricSpecifications?: PLM_FabricSpecification[] | null
  } | null
}

export interface PlmExportWashQueryData {
  exportWash: { requestId: string }
}

// ── Mutations ──────────────────────────────────────────────────────────────────

export const PLM_ADD_WASH: TypedDocumentNode<PlmAddWashMutationData, PlmAddWashMutationVariables> = gql`
  mutation plmAddWash($wash: WashInput!, $includeFabricData: Boolean = false, $includeFabricSpecData: Boolean = false) {
    addWash(wash: $wash) {
      ...PlmWashDetailsFragment
    }
  }
  ${PLM_WASH_DETAILS_FRAGMENT}
`

export const PLM_UPDATE_WASH: TypedDocumentNode<PlmUpdateWashMutationData, PlmUpdateWashMutationVariables> = gql`
  mutation plmUpdateWash($updateWash: WashInput!, $includeFabricData: Boolean = false, $includeFabricSpecData: Boolean = false) {
    updateWash(updateWash: $updateWash) {
      ...PlmWashDetailsFragment
    }
  }
  ${PLM_WASH_DETAILS_FRAGMENT}
`

export const PLM_UPDATE_WASH_TEAMS: TypedDocumentNode<PlmUpdateWashTeamsMutationData, PlmUpdateWashTeamsMutationVariables> = gql`
  mutation plmUpdateWashTeams($id: ID!, $teamsToUpdate: TeamsToUpdateInput) {
    updateWashTeams(id: $id, teamsToUpdate: $teamsToUpdate)
  }
`
