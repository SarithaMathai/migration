export const ADD_PAPER = gql`
  mutation addPaper($material: SPARK_PaperInput!) {
    addMaterial: addPaper(material: $material) {
      ...PaperDetails
    }
  }
  ${PAPER_DETAILS}
`

export const UPDATE_PAPER = gql`
  mutation updatePaper($material: SPARK_PaperInput!) {
    updatePaper(material: $material) {
      ...PaperDetails
    }
  }
  ${PAPER_DETAILS}
`

export const GET_PAPER_BY_ID = gql`
  query getHubMaterial($id: ID!) {
    getHubMaterial(id: $id) {
      ... on SPARK_Paper {
        ...PaperDetails
      }
    }
    getUserAccessUnencoded(resourceIds: [$id]) {
      resourcePermissions {
        permissions
      }
    }
  }
  ${PAPER_DETAILS}
`
