import { gql } from '@apollo/client'

export const GET_RESOURCE_TRAINING_LABELS = gql`
  query getTrainingLabels {
    getTrainingLabels {
      id
      name
    }
  }
`

export const SEARCH_RESOURCE_TRAINING_DOCUMENTS = gql`
  query searchTrainingDocuments($topics: [String]) {
    searchTrainingDocuments(topics: $topics) {
      id
      title
      createdBy
      createdDate
      updatedDate
      attachment {
        id
        type
        title
        extensions {
          mediaType
          fileSize
        }
      }
      labels {
        id
        name
      }
      content
    }
  }
`

export const GET_ALL_RESOURCE_TRAINING_DOCUMENTS = gql`
  query getAllTrainingDocuments($userType: String) {
    getAllTrainingDocuments {
      labels {
        id
        name
      }
      documents(userType: $userType) {
        searchResults {
          id
          title
          content
          labels {
            id
            name
          }
          updatedDate
          createdDate
          tags
          topics {
            name
            isIgnore
          }
          verificationGroup
          contentTypes
          attachments {
            attachmentId
            attachmentName
            extensions {
              fileSize
              mediaType
            }
          }
          updatedBy {
            firstName
            lastName
            isInternal
          }
          createdBy {
            firstName
            lastName
            isInternal
          }
        }
      }
    }
  }
`

export const GET_IMAGE = gql`
