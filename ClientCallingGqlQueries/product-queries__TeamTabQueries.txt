        ...AttachmentsWithMetaData

      createdBy {
        id

          id
        }
      }
      teams(includeSystemTeams: true) {
        content {
          id
          name
          partner_name
          partner_id
          business_partner {
            id
            name
            role {
              bpRoleId
              bpRoleName
              roleCategory
            }
          }
          accessControlGroupIds
          teamType
          members {
            user_profile_id
            roles
          }
          status_name
          partnerType {
            code
            description
          }
        }
      }
      businessPartners {
        id
        name
        associatedPartnerType
        role {
          bpRoleName
          bpRoleId
          status {
            id
          }
        }
      }
      droppedPartnerIds
      droppedPartners {
        id
        name
        associatedPartnerType
        role {
          bpRoleName
          bpRoleId
          status {
            id
          }
        }
      }
      notRemovablePartnerIds
      resources {
        mappings {
          team
        }
      }
      discussionsV2 {
        content {
          discussionDto {
            id
            partnerId
          }
        }
      }

        code
        description
      }
      unDroppablePartners(isDesignPartner: $isDesignPartner)
    }
  }
  ${ATTACHMENTS_WITH_META_DATA_FRAGMENT}

export const GET_PRODUCT_WITH_TEAMS = gql`
  query getProductWithTeams($id: ID!, $partnerId: String, $workspaceId: String, $isDesignPartner: Boolean) {
    getProduct(id: $id) {
      id
      attachmentsV3 {
        content {
          id
        }
      }
      createdBy {
        id
        currentVMMRole
        businessPartner {
          ...VmmBusinessPartnerOnProduct
        }
      }
      teams(partnerId: $partnerId, workspaceId: $workspaceId) {
        content {
          ...ProductFullTeamFragment
        }
        paging {
          totalElements
        }
      }
      unfilteredTeams: teams {
        paging {
          totalElements
        }
      }
      businessPartners {
        ...VmmBusinessPartnerOnProduct
      }
      notRemovablePartnerIds
      droppedPartnerIds
      unDroppablePartners(isDesignPartner: $isDesignPartner)
      droppedPartners {
        ...VmmBusinessPartnerOnProduct
      }
      resources {
        mappings {
          team
        }
      }
      productWorkspaceInfo {
        workspaceId
        teams
        designCycle {
          id
          name
        }
        humanId
        setDates
        protoApprovalDueDate
        carryForwardReason
        linkedAttributes
      }
    }
  }
  ${VMM_BUSINESS_PARTNER_ON_PRODUCT}
  ${PRODUCT_FULL_TEAM_FRAGMENT}
`
