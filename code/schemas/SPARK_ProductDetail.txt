    createProductDetailsSet(productDetails: productDetailSetInput): SPARK_ProductDetails
    updateProductDetailAccess(updateAccessPermissions: SPARK_ProductDetailsManagePermissionInput): [SPARK_ProductDetails]

    cloneFilesForProductDetails(attachmentIds: [String], cloneReference: [PDTLAttachmentCloneRef]): [SPARK_Attachment]

    workspaceContext: [String]
