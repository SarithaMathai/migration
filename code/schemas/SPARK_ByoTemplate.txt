extend type Query {
	getByoTemplates(userAttributes: SPARK_UserProfileAttributesInput): [SPARK_ByoTemplate]
	getByoTemplateById(id: ID!): SPARK_ByoTemplate
}

	addByoTemplate(sparkByoTemplate: SPARK_ByoTemplateInput): SPARK_ByoTemplate

	shareByoTemplate(id: ID!, userProfileIds: [String]): [SPARK_ByoTemplate]

	connectedProducts: Boolean

	colId: String,
	hide: Boolean,

	sortIndex: String,
	rowGroup: Boolean,
	rowGroupIndex: String,
}

	key: String
	value: SPARK_ByoGridFilterValue
}
input SPARK_ByoGridFilterValue {
	filterType: String,
	type: String,
	filter: String,
	operator: String,
	condition1: SPARK_GridFilterType
	condition2: SPARK_GridFilterType
}

input SPARK_GridFilterType {
	filterType: String
	type: String
	filter: String
}

	key: String
	value: SPARK_ByoGridFiltersValue
}
type SPARK_ByoGridFiltersValue {
	filterType: String,
	type: String,
	filter: String,
	operator: String,
	condition1: SPARK_GridFiltersType

type SPARK_GridFiltersType {

	filter: String

  rowGroup: Boolean,
  rowGroupIndex: Int,
