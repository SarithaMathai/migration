extend type Query {
  getToDos: [SPARK_ToDo]
}

extend type Mutation {
  deleteToDoByBusinessPartner(
    resourceType: String!
    partnerId: ID!
    resourceIds: [ID]
  ): ID
  addToDo(toDo: SPARK_ToDoInput!): SPARK_ToDo

type SPARK_ToDo {

  userProfileId: String!

  partnerId: ID
}

type SPARK_ToDoResource {
  id: ID!
  type: String!
  resourceDescription: String!
  thumbnailId: String
  relatedResource: String
}

input SPARK_ToDoInput {
  description: String!
  resource: SPARK_SingleResourceMappingInput
}
