  Mutation: {
    validateUniqueness: async (parent, { sparkTrim }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        'SPARK-TRIM-INTERNAL-TEAM',
        ctx,
      )
      const uniqunessResult = await ctx.loaders.trim.validateUniqueness(permissionJWT).load(sparkTrim)
      return uniqunessResult
    },
    addTrim: async (parent, { sparkTrim }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        'SPARK-TRIM-INTERNAL-TEAM',
        ctx,
      )
      const trim = await ctx.loaders.trim.addTrim(permissionJWT).load(sparkTrim)
      // Setup relationships
      if (sparkTrim.files && sparkTrim.files.length > 0) {
        const associateAttachmentTeamDto = { 
          teamsToUpdateDto: { teamsToAdd: sparkTrim.teamsToAdd }, 
          parentId: trim.humanId, 
          humanIds: sparkTrim.files,
          relatedResourceIds: [trim.humanId],
        }
        const attachmentPermissionJWT = await getUserPermissionsJWT(sparkTrim.files, ctx)
        await ctx.loaders.attachment.associateAttachmentTeams(attachmentPermissionJWT).load(associateAttachmentTeamDto, { humanId: trim.humanId })
      }
      return trim
    },
    updateTrim: async (parent, { sparkTrim }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(sparkTrim.humanId, ctx)
      return ctx.loaders.trim.updateTrim(permissionJWT).load(sparkTrim)
    },
    updateTrimTeams: async (parent, { humanId, teamsToUpdate }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(humanId, ctx)
      return ctx.loaders.trim.updateTrimTeams(permissionJWT).load(teamsToUpdate, { humanId })
    },
    updateTrimVariations: async (parent, { sparkTrimVariations }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(sparkTrimVariations.humanId, ctx)
      return ctx.loaders.trim.updateTrimVariations(permissionJWT).load(sparkTrimVariations)
    },
  },

  SPARK_Trim: {
    createdBy: (trim, args, ctx) => getUser(trim.createdBy, ctx),
    updatedBy: (trim, args, ctx) => getUser(trim.updatedBy, ctx),
    designCycle: async (trim, args, ctx) => {
      const designCycleId = (trim && trim.designCycle) || null
      const response = (designCycleId && await loadTags([designCycleId], ctx))
      return response && response.length > 0 ? response[0] : null
    },

    bomSizeOptions: (trim) => getSizeOptions(trim),
    bomUomOptions: (trim) => getBomUomOptions(trim),
    baseSizeCaption: (trim) => getBomSizeCaption(trim),
    brand: async (trim, args, ctx) => {
      const brandData = trim.brandId ? await ctx.loaders.brand.getBrand.load(trim.brandId) : null

        return {
          id: brandData.brandId,
          code: brandData.brandId,
          name: brandData.brandName,
          description: brandData.brandName,
        }
      }
    },
  },
