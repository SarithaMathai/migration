import _ from 'lodash'
import {getUserPermissionsJWT, getUserPermissionsJWTHelper} from '../utils/commonLoaders'
import {DEFAULT_SAMPLE_CHUNK_SIZE, SAMPLE_EVALUTION} from '../utils/constants'
import {loadBps, loadBpWithType} from '../utils/vmmUtils'
import SPARK_WorkspaceV2 from './SPARK_WorkspaceV2'
import {bulkEvaluateSampleUtil} from '../utils/sampleUtils'
import {createSampleAttachmentRelationship} from '../utils/discussionUtils'
import {isColorId} from '../utils/materials/colorUtils'
import SPARK_Material from './SPARK_Material'
import {RESOURCE_TYPES_PREFIXES} from '../config/constants'
import {getParticipantsByUserGroup} from '../utils/userGroupUtils'
import {systemUser} from '../utils/userAttributesUtils'
import {batchParallelOperation} from '../utils/batchingUtils'

export default {
  Query: {
    getSampleById: async (parent, {id}, ctx) => {
      const samples = await ctx.loaders.sampleV2
        .getSampleById
        .load(id)
      return samples
    },
    getSamplesByIdsV2: async (parent, {ids, recentlyViewed = true}, ctx) => {
      // Batch loading of samples by ids
      const operation = async (batchIds) => {
        const permissionJWT = await getUserPermissionsJWTHelper(batchIds, ctx)
        return ctx.loaders.sampleV2
          .getSamplesByIdsV2ByPost(permissionJWT)
          .load(batchIds)
      }
      const samples = await batchParallelOperation({batchSize: DEFAULT_SAMPLE_CHUNK_SIZE, operation, items: ids})
      if (samples && samples.length === 1 && recentlyViewed) {
        ctx.loaders.recentlyViewed.addRecentlyViewed.load({
          id: samples[0].id,
          type: 'sample',
        })
      }
      return samples
    },
    getSampleMaterialTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getSampleMaterialTypesV2.load(),
    getSampleTypesV2: (parent, {resourceTypes}, ctx) =>
      ctx.loaders.sampleV2.getSampleTypesV2.load(resourceTypes),
    getSampleProductTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getSampleProductTypesV2.load(),
    getSampleTrackingTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getSampleTrackingTypesV2.load(),
    getSampleLateReasonTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getSampleLateReasonTypesV2.load(),
    getColorSamplePurposesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getColorSamplePurposesV2.load(),
    getFabricSampleTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getFabricSampleTypesV2.load(),
    getProductSampleEvaluationTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getProductSampleEvaluationTypesV2.load(),
    getMaterialSampleEvaluationTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getMaterialSampleEvaluationTypesV2.load(),
    getSampleRounds: async (parent, {humanId}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(humanId, ctx)
      return ctx.loaders.sampleV2.getSampleRounds(permissionJWT).load(humanId)
    },
    getSamplesByParentId: async (parent, {humanId}, ctx) => {
      const sampleRelationship = await ctx.loaders.relationship.getByID.load({
        id: humanId,
        type: 'sample',
        maxDepth: 0,
      })
      if (sampleRelationship.nodes && sampleRelationship.nodes.length > 0) {
        const permissionJWT = await getUserPermissionsJWT(
          sampleRelationship.nodes.map(sample => sample.id),
          ctx,
        )
        return ctx.loaders.sampleV2
          .getSamplesByIdsV2(permissionJWT)
          .load(sampleRelationship.nodes.map(sample => sample.id))
      } else return []
    },
    getColorSamplesByParentId: async (parent, {id}, ctx) =>
      ctx.loaders.sampleV2.getColorSamplesByParentId.load(id),
    getSampleLocationByIds: async (parent, {ids}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(ids, ctx)

      const operation = async (batchIds) => {
        return ctx.loaders.sampleV2.getSamplesByIdsV2(permissionJWT).load(batchIds)
      }
      const samples = await batchParallelOperation({batchSize: DEFAULT_SAMPLE_CHUNK_SIZE, operation, items: ids})
      if (!Array.isArray(samples)) return []

      const response = await Promise.all(
        samples.map(async (sample) => {
          if (!sample?.rfidTagIds?.length) return []

          const latestRfidLocationsResponse = await ctx.loaders.search.searchLatestRfidLocations.load({q: sample.rfidTagIds.join(" OR ")})
          const locations = Array.from(latestRfidLocationsResponse ?? [])
          if (!locations.length) return []

          const latestLocation = locations.reduce((previous, next) => {
            const previousLastSeenTs = Number.isFinite(Date.parse(previous.lastSeen)) ? Date.parse(previous.lastSeen) : 0
            const nextLastSeenTs = Number.isFinite(Date.parse(next.lastSeen)) ? Date.parse(next.lastSeen) : 0
            return nextLastSeenTs > previousLastSeenTs ? next : previous
          })
          return [{
            id: sample.id,
            locationDescription: latestLocation.locationDescription,
            lastSeen: latestLocation.lastSeen,
          }]
        }))
      return _.flatten(response)
    },
    getSamplesByRfidTagIds: async (parent, {ids}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(ids, ctx)
      return ctx.loaders.sampleV2
        .getSamplesByRfidTagIds(permissionJWT)
        .load(ids)
    },
    getSampleNotificationErrors: (parent, {}, ctx) => ctx.loaders.notification.getSampleNotificationErrors.load(),
    getSampleExports: (parent, {}, ctx) => ctx.loaders.sampleV2.getSampleExports.load(),
    getWashSampleTypesV2: (parent, {}, ctx) =>
      ctx.loaders.sampleV2.getWashSampleTypesV2.load(),
    getSampleFormats: (parent, {type}, ctx) => ctx.loaders.sampleV2.getSampleFormats.load(type),
    getMaterialSampleFormats: (parent, {type}, ctx) => ctx.loaders.sampleV2.getMaterialSampleFormats.load(type),
    getSampleEvaluationPurposes: (parent, {}, ctx) => ctx.loaders.sampleV2.getSampleEvaluationPurposes.load(),
    getSampleTypeFormatMappings: (parent, {}, ctx) => ctx.loaders.sampleV2.getSampleTypeFormatMappings.load(),
  },

  Mutation: {
    createSamplesV2: async (parent, {samples}, ctx) => {
      const samplePayload = await ctx.loaders.sampleV2.createSamplesV2.load(samples)

      const sampleRecord = samples && samples.newSamples && samples.newSamples[0]
      if (sampleRecord && sampleRecord.files && sampleRecord.files.length > 0) {
        // V2 File Upload and Adding Relationship
        const sampleHumanId = samplePayload && samplePayload[0].id
        await createSampleAttachmentRelationship(ctx, sampleHumanId, sampleRecord.files)

        const permissionJWT = await getUserPermissionsJWT(
          sampleRecord.files,
          ctx,
        )

        const relatedResources = [sampleRecord.parentHumanID, sampleHumanId]
        const attachments = sampleRecord.files.map(id => ({
          documentId: id,
          relatedResources: relatedResources,
          resource: {
            id: sampleHumanId,
            type: 'sample',
          },
        }))
        ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load({attachments: attachments})
      }
      return samplePayload
    },
    createSampleRoundV2: async (parent, {sample, sampleId}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        [sampleId, SAMPLE_EVALUTION],
        ctx,
      )
      return ctx.loaders.sampleV2
        .createSampleRoundV2(permissionJWT)
        .load(sample, {sampleId})
    },
    bulkEvaluateSamples: async (parent, {
      updateSamples,
      newSampleRounds,
    }, ctx) => bulkEvaluateSampleUtil(ctx, updateSamples, newSampleRounds),
    updateSamplesV2: async (parent, {samples}, ctx) => {
      const sampleIds = samples.updateSamples &&
        samples.updateSamples.length > 0 &&
        samples.updateSamples.map(sample => sample.id)
      const permissionJWT = await getUserPermissionsJWTHelper(
        sampleIds.concat(SAMPLE_EVALUTION),
        ctx,
      )
      return ctx.loaders.sampleV2.updateSamplesV2(permissionJWT).load(samples)
    },
    updateSampleWorkspaceAssociation: async (parent, {sampleId, workspaceId}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT([sampleId, workspaceId], ctx)
      return ctx.loaders.sampleV2.updateSampleWorkspaceAssociation(permissionJWT, sampleId, workspaceId).load()
    },
    retrySampleNotificationError: (parent, {failedMessageId}, ctx) =>
      ctx.loaders.notification.retrySampleNotificationError.load(null, failedMessageId),
    retryAllSampleNotificationErrors: (parent, {}, ctx) =>
      ctx.loaders.notification.retryAllSampleNotificationErrors.load(),
    requestSampleExport: (parent, {}, ctx) => ctx.loaders.sampleV2.requestSampleExport.load(),
    bulkCloneFilesForEvaluate: async (parent, {attachmentIds, cloneReferences}, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachmentIds, ctx)

      const response = await Promise.all(
        attachmentIds.map(async (id) => {
          const clones = await ctx.loaders.attachment.cloneAttachmentV3(permissionJWT)
            .load({cloneReferences}, id)
          return clones
        }))

      return _.flatten(response)
    },
  },

  SPARK_SampleAsset: {
    __resolveType(data) {
      if (isColorId(data.humanId)) {
        return 'SPARK_Color'
      } else if (data.humanId.startsWith(RESOURCE_TYPES_PREFIXES.Artworks)) {
        return 'SPARK_Artwork'
      }
      return null
    },
  },

  SPARK_SampleV2: {
    createdBy: (sample, args, ctx) => {
      if (sample.systemGenerated) {
        return systemUser(sample.createdBy)
      }
      return ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(sample.createdBy)
    },
    evaluatedBy: (sample, args, ctx) =>
      sample.evaluatedBy && ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(sample.evaluatedBy),
    updatedBy: (sample, args, ctx) => {
      if (sample.systemGenerated && sample.createdBy === sample.updatedBy) {
        return systemUser(sample.updatedBy)
      }
      return ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(sample.updatedBy)
    },
    product: (sample, args, ctx) =>
      (sample.parentId && sample.parentId.indexOf('PID') === 0)
        ? ctx.loaders.product.getByID.load(sample.parentId)
        : null,
    workspace: async (sample, args, ctx) => {
      const workspaceId = sample.workspaceContext
      if (workspaceId) {
        const workspaceData = await SPARK_WorkspaceV2.Query.getWorkspaceV2({}, {id: workspaceId}, ctx)
        return _.isEmpty(workspaceData) ? null : workspaceData
      }
      return null
    },
    designPartnerId: sample => {
      return sample.dpPartnerId ? [sample.dpPartnerId] : []
    },
    trim: async (sample, args, ctx) => {
      if (sample && sample.parentId) {
        const permissionJWT = await getUserPermissionsJWT(sample.parentId, ctx)
        return ctx.loaders.trim.getTrim(permissionJWT).load(sample.parentId)
      }
      return null
    },
    colorArchroma: async (sample, args, ctx) => {
      if (sample.parentId && (sample.parentId.indexOf('ARCCLR') === 0 || sample.parentId.indexOf('TARARCCLR') === 0 || sample.parentId.indexOf('REFARCCLR') === 0)) {
        const permissionJWT = await getUserPermissionsJWT(sample.parentId, ctx)
        return ctx.loaders.colorArchroma.getByID(permissionJWT).load(sample.parentId)
      } else {
        return null
      }
    },
    fabricSpecCombo: async (sample, args, ctx) => {
      if (sample.parentId && sample.parentId.indexOf('FSC') === 0) {
        const permissionJwt = await getUserPermissionsJWT(sample.parentId, ctx)
        return ctx.loaders.combination.getFabricSpecComboById(permissionJwt).load(sample.parentId)
      } else {
        return null
      }
    },
    fabricSpec: async (sample, args, ctx) => {
      if (sample.parentId && sample.parentId.indexOf('FAS') === 0) {
        return ctx.loaders.fabric.getSpecificationByID.load(sample.parentId)
      } else {
        return null
      }
    },
    businessPartner: (sample, args, ctx) =>
      sample.partnerId && loadBpWithType(ctx, sample.partnerId),
    fabricSupplier: (sample, args, ctx) =>
      sample.fabricSupplierId && ctx.loaders.vmm.getByID.load(sample.fabricSupplierId),
    merchandiseVendors: async (sample, args, ctx) =>
      (sample.merchandiseVendorIds && sample.merchandiseVendorIds.length > 0) ? loadBps(ctx, sample.merchandiseVendorIds) : [],
    discussionParticipants: sample => {
      if (
        !sample.discussionParticipants ||
        _.isEmpty(sample.discussionParticipants)
      ) {
        return {teams: [], users: []}
      }
      return sample.discussionParticipants
    },
    participants: (sample, args, ctx) => {
      if (!args.isParticipantsFromUserGroup) {
        if (
          !sample.discussionParticipants ||
          _.isEmpty(sample.discussionParticipants)
        ) {
          return getParticipantsByUserGroup(sample.id, ctx)
        }
        return sample.discussionParticipants
      } else {
        return getParticipantsByUserGroup(sample.id, ctx)
      }
    },
    artwork: async (sample, args, ctx) => {
      const assetHumanId = sample.assetHumanId
      if (assetHumanId && assetHumanId.startsWith(RESOURCE_TYPES_PREFIXES.Artworks)) {
        const permissionJWT = await getUserPermissionsJWT(assetHumanId, ctx)
        return ctx.loaders.artwork.getByID(permissionJWT).load(assetHumanId)
      }
      return null
    },
    attachments: async (sample, args, ctx) => {
      const attachments = await ctx.loaders.search.searchAttachmentsByRelatedResource.load(sample.id)
      if (!attachments || attachments.length < 1) return []
      return attachments
    },
    rfidLocationInfo: async (sample, args, ctx) => {
      if (!Array.isArray(sample.rfidTagIds) || sample.rfidTagIds.length === 0) return null

      const latestRfidLocationsResponse = await ctx.loaders.search.searchLatestRfidLocations.load({q: sample.rfidTagIds.join(" OR ")})
      const locations = Array.from(latestRfidLocationsResponse ?? [])
      if (!locations.length) return null

      const latestLocation = locations.reduce((previous, next) => {
        const previousLastSeenTs = Number.isFinite(Date.parse(previous.lastSeen)) ? Date.parse(previous.lastSeen) : 0
        const nextLastSeenTs = Number.isFinite(Date.parse(next.lastSeen)) ? Date.parse(next.lastSeen) : 0
        return nextLastSeenTs > previousLastSeenTs ? next : previous
      })
      return latestLocation.locationDescription
    },
    sampleMeasurementSetAssociation: (sample) => {
      return sample.sampleMeasurementSet
    },
    sampleMeasurementSet: async (sample, args, ctx) => {
      if (sample.sampleMeasurementSet && sample.sampleMeasurementSet.sampleMeasurementSetId) {
        const permissionJWT = await getUserPermissionsJWT([sample.id, sample.sampleMeasurementSet.sampleMeasurementSetId], ctx)
        const measurementSet = ctx.loaders.measurement.getSampleMeasurement(permissionJWT, sample.id)
        return measurementSet
      } else {
        return null
      }
    },
    createdByInternalPrimaryRole: async ({createdByInternalPrimaryRole}, args, ctx) => {
      const userRoles = await ctx.loaders.role.getPage.load(createdByInternalPrimaryRole)
      if (userRoles && userRoles[createdByInternalPrimaryRole]) {
        return {
          id: createdByInternalPrimaryRole,
          value: userRoles[createdByInternalPrimaryRole],
        }
      } else {
        return null
      }
    },
    evaluatedByInternalPrimaryRole: async ({evaluatedByInternalPrimaryRole}, args, ctx) => {
      const userRoles = await ctx.loaders.role.getPage.load(evaluatedByInternalPrimaryRole)
      if (userRoles && userRoles[evaluatedByInternalPrimaryRole]) {
        return {id: evaluatedByInternalPrimaryRole, value: userRoles[evaluatedByInternalPrimaryRole]}
      } else {
        return null
      }
    },
    designCycle: async (sample, args, ctx) => {
      if (sample && sample.designCycle) {
        return ctx.loaders.tag.getTag.load(sample.designCycle)
      } else {
        return null
      }
    },
    designEvaluators: (sample, args, ctx) =>
      _.map(sample.designEvaluators, evaluator => ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(evaluator)),
    technicalEvaluators: (sample, args, ctx) =>
      _.map(sample.technicalEvaluators, evaluator => ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(evaluator)),
    asset: (sample, args, ctx) => {
      if (sample && sample.assetHumanId) {
        return SPARK_Material.Query.getMaterialByIdFromService(null, {id: sample.assetHumanId}, ctx)
      }
      return null
    },
    clmPackage: (sample, args, ctx) => {
      if (sample && sample.clmPackageIdentifier) {
        return ctx.loaders.tgtColorEvaluator.getClmPackageById.load(sample.clmPackageIdentifier)
      }
      return null
    },
    brand: (sample, args, ctx) =>
      sample.brandId ? ctx.loaders.brand.getBrand.load(sample.brandId) : null,
    currentLocations: async (sample, args, ctx) => {
      if (!Array.isArray(sample.rfidTagIds) || sample.rfidTagIds.length === 0) return []
      const response = await ctx.loaders.search.searchLatestRfidLocations.load({q: sample.rfidTagIds.join(" OR ")})
      return Array.from(response ?? []).map(latestSampleRfidLocation => ({
        epc: latestSampleRfidLocation.epc,
        locationDescription: latestSampleRfidLocation.locationDescription,
        lastSeen: latestSampleRfidLocation.lastSeen,
      }))
    },
  },

  SPARK_SampleDepartment: {
    department: async (parent, args, ctx) =>
      (await (parent.id &&
        parent.id !== -1 &&
        ctx.loaders.ig.department.getByID.load(parent.id))) || null,
  },

  // TODO remove this resolver, just use the normal bp resolver...
  SPARK_SampleDiscussionParticipantTeamInfoV2: {
    businessPartner: (team, args, ctx) => {
      if (team.businessPartnerId === 0 || team.businessPartnerId === null) {
        return {bpId: 0, bpName: 'Target'}
      }
      return ctx.loaders.vmm.getByID.load(team.businessPartnerId)
    },
  },

  SPARK_SampleDiscussionParticipantUserInfoV2: {
    userDetails: async (member, args, ctx) => {
      if (member.systemUser) {
        return systemUser(member.userId)
      } else {
        return ctx.loaders.userAttributes.getUserByID.load(member.userId)
      }
    },
  },
  SPARK_SampleDiscussionsParticipantsV2: {
    teams: participants => participants.teams,
    users: participants => participants.users,
  },

  SPARK_SampleLibraryColorsV2: {
    color: async (sampleLibraryColor, args, ctx) => {
      const token = await getUserPermissionsJWT(sampleLibraryColor.id, ctx)
      return ctx.loaders.color.getByID(token).load(sampleLibraryColor.id)
    },
  },
}
