import { getUserPermissionsJWT } from '../../utils/commonLoaders'

const discussionActivityModifiedDataHelper = async (activity, streamResourceId, ctx) => {
  if (!ctx) return []
  const {
    loaders: {
      discussion: { getDiscussionsV2 } = {},
    } = {},
  } = ctx

  const {
    resourceType = '',
    resourceId = '',
    eventType = 'UPDATE',
    eventSummary: { description = '' } = {},
  } = activity || {}

  const permissionJWT = await getUserPermissionsJWT([resourceId], ctx)
  const discussionData = await getDiscussionsV2(permissionJWT).load({ ids: [resourceId] })

  const productId = discussionData?.[0]?.resources?.mappings?.product?.[0]
  const workspaceId = discussionData?.[0]?.resources?.mappings?.workspace?.[0]

  const parentResourceInfo = {
    linkGenerationVariables: { discussionId: resourceId, productId, ...(workspaceId && { workspaceId }) },
  }

  return [{
    attributeUpdateText: '',
    resourceUpdated: {
      resourceType: resourceType,
      resourceId: resourceId,
      resourceDescription: description,
      parentResourceInfo,
    },
    resourceChangeTypeText: `was ${String(eventType).toLowerCase()}d by`,
  }]
}

export default discussionActivityModifiedDataHelper
