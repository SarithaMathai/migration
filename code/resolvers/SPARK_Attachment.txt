import _ from 'lodash'
import {
  getUserPermissionsJWT,
} from '../utils/commonLoaders'
import { get3DmodelFile } from '../utils/Product/attachmentUtils'
import { getUser } from '../utils/userAttributesUtils'

export default {
  Query: {
    getAttachmentsV3: async (parent, { ids }, ctx) => {
      if (!ids || ids.length === 0) {
        return []
      }
      const permissionJWT = await getUserPermissionsJWT(ids, ctx)
      return ctx.loaders.attachment.getAttachmentsV3(permissionJWT).load(ids)
    },
    getAttachmentsByResource: async (parent, { resourceId }, ctx) => {
      let attachments = []
      const attachmentsIds = await ctx.loaders.relationship.searchByIds.load({
        id: resourceId,
        includeNodeTypes: ['attachments', 'attachments_v3'],
        maxDepth: 0,
      })

      if (attachmentsIds.length && !_.isEmpty(attachmentsIds[0].nodes)) {
        const aclPermissionJWT = await ctx.loaders.accessControl.getUserAccessByPost.load(
          attachmentsIds[0].nodes.map(node => node.id),
        )
        attachments = await ctx.loaders.attachment
          .getAttachmentsByPostV3(aclPermissionJWT)
          .load(attachmentsIds[0].nodes.map(node => node.id))
      }
      return attachments
    },
    getAttachmentsByResourceAndOwner: async (parent, { resourceId }, ctx) => {
      let attachments = []
      const attachmentsIds = await ctx.loaders.relationship.searchByIds.load({
        id: resourceId,
        includeNodeTypes: ['attachments', 'attachments_v3'],
        maxDepth: 0,
      })

      if (attachmentsIds.length && !_.isEmpty(attachmentsIds[0].nodes)) {
        const attSearch = { ids: attachmentsIds[0].nodes.map(node => node.id) }
        attachments = await ctx.loaders.attachment
          .getAttachmentsByIdsAndAuthorByPostV3()
          .load(attSearch)
      }
      return attachments
    },
    getRendersForAttachmentIds: async (parent, { ids, betaMode }, ctx) => {
      const allAttachments = await Promise.all(ids.map(async (id) => {
        return ctx.loaders.attachment.getRendersForAttachmentIds(betaMode).load(id)
      }))
      return _.compact(allAttachments)
    },
    getRendersForAttachmentV3Ids: async (parent, { ids, betaMode }, ctx) => {
      const allAttachments = await Promise.all(ids.map(async (id) => {
        return ctx.loaders.attachment.getRendersForAttachmentV3Ids(betaMode).load(id)
      }))
      return _.compact(allAttachments)
    },
    getRendersForAttachmentIdsByPost: async (parent, { ids, betaMode }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(ids, ctx)
      return ctx.loaders.attachment.getRendersForAttachmentIdsByPost(permissionJWT).load(ids, { 'betaMode': betaMode })
    },
    getAttachmentsfromRelatedResource: async (parent, { relatedResourceId, parentResourceId }, ctx) => {
      let attachments
      if (parentResourceId && relatedResourceId) {
        attachments = await ctx.loaders.search.searchAttachmentsByParentAndRelatedResource.load({ parentResourceId, relatedResourceId })
      } else if (relatedResourceId) {
        attachments = await ctx.loaders.search.searchAttachmentsByRelatedResource.load(relatedResourceId)
      }
      if (attachments && attachments.length) {
        return attachments
      } else {
        return []
      }
    },
    getAttachmentsfromRelatedResources: async (parent, { relatedResourceIds }, ctx) => {
      let attachments = await ctx.loaders.search.searchAttachmentsByRelatedResources.load({ resourceIds: relatedResourceIds })
      if (attachments && attachments.content.length) {
        return attachments.content
      } else {
        return []
      }
    },
  },

  Mutation: {
    archiveAttachmentV3: async (parent, { id }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(id, ctx)
      return ctx.loaders.attachment.archiveAttachmentV3(permissionJWT).load(id)
    },
    deleteAttachmentV3: async (parent, { humanId }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(humanId, ctx)
      return ctx.loaders.attachment.deleteAttachmentV3(permissionJWT).load(humanId)
    },
    copyAttachmentsV3: async (parent, { attachmentInput }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        attachmentInput.humanIds,
        ctx,
      )
      return ctx.loaders.attachment
        .copyAttachmentsV3(permissionJWT)
        .load(attachmentInput)
    },
    associateResourcesV3: async (parent, { associateAttachmentInput }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        associateAttachmentInput.humanIds,
        ctx,
      )
      return ctx.loaders.attachment
        .associateResourcesV3(permissionJWT)
        .load(associateAttachmentInput)
    },
    removeResourcesV3: async (parent, { associateAttachmentInput }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(
        associateAttachmentInput.humanIds,
        ctx,
      )
      return ctx.loaders.attachment
        .removeResourcesV3(permissionJWT)
        .load(associateAttachmentInput)
    },
    updateAttachmentsACLPermissions: async (parent, { partnerId, attachmentIdsForAdmin, attachmentsIdsForRead }, ctx) => {
      let bulkAddACLAccessDTOList = []
      attachmentIdsForAdmin.forEach((attachmentId) => {
        bulkAddACLAccessDTOList.push({
          resourceId: attachmentId,
          dps: [{
            permissionLevel: 'ADMIN',
            grantees: [partnerId],
          }],
        })
      })
      attachmentsIdsForRead.forEach((attachmentId) => {
        bulkAddACLAccessDTOList.push({
          resourceId: attachmentId,
          dps: [{
            permissionLevel: 'READ',
            grantees: [partnerId],
          }],
        })
      })
      const accessControl = await ctx.loaders.accessControl.updateAccessControl.load(bulkAddACLAccessDTOList)
      return accessControl
    },

    updateAttachmentTags: async (parent, { attachmentId, tags }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
      return ctx.loaders.attachment.updateTagsV3(permissionJWT).load({ attachmentId, tags })
    },
    updateAttachmentTagsV3: async (parent, { attachmentId, tags }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
      return ctx.loaders.attachment.updateTagsV3(permissionJWT).load({ attachmentId, tags })
    },
    bulkUpdateAttachments: async (parent, { tags, permissions }, ctx) => {
      if (!_.isEmpty(tags)) {
        const attachmentIds = tags.map(tag => tag.id)
        const permissionJWT = await getUserPermissionsJWT(attachmentIds, ctx)
        ctx.loaders.attachment.bulkUpdateTags(permissionJWT).load(tags)
      }
      if (permissions) {
        ctx.loaders.accessControl.bulkUpdateAttachmentPermissions.load(permissions)
      }
    },

    updateAttributes: async (parent, { attachment }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachment.documentId, ctx)
      return ctx.loaders.attachment.updateAttributes(permissionJWT).load(attachment)
    },

    bulkUpdateAttributes: async (parent, { attachments }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachments.attachments.map(attachment => attachment.documentId || attachment.humanId), ctx)
      return ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load(attachments)
    },
    bulkUpdateAttachmentsV2: async (parent, { attachments }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(attachments.map(attachment => attachment.documentId), ctx)
      return ctx.loaders.attachment.bulkUpdateAttachmentsV2(permissionJWT).load({ attachments })
    },
    publishAttachmentToGallery: async (parent, { attachmentId }, ctx) => {
      const isAttachmentV3 = attachmentId && attachmentId.startsWith('ATC-')
      if (isAttachmentV3) {
        const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
        // api returns void, given the assumption if something went wrong we get an error, we can return true with no error
        await ctx.loaders.attachment.publishAttachmentToGalleryV3(permissionJWT).load(null, attachmentId)
        return true
      } else {
        const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
        // api returns void, given the assumption if something went wrong we get an error, we can return true with no error
        await ctx.loaders.attachment.publishAttachmentToGallery(permissionJWT).load(null, attachmentId)
        return true
      }
    },
    unpublishAttachmentToGallery: async (parent, { attachmentId }, ctx) => {
      const isAttachmentV3 = attachmentId && attachmentId.startsWith('ATC-')
      if (isAttachmentV3) {
        const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
        // api returns void, given the assumption if something went wrong we get an error, we can return true with no error
        await ctx.loaders.attachment.unpublishAttachmentToGalleryV3(permissionJWT).load(null, attachmentId)
        return true
      } else {
        const permissionJWT = await getUserPermissionsJWT(attachmentId, ctx)
        // api returns void, given the assumption if something went wrong we get an error, we can return true with no error
        await ctx.loaders.attachment.unpublishAttachmentToGallery(permissionJWT).load(null, attachmentId)
        return true
      }
    },
    associateAttachmentTeams: async (parent, { parentId, files, teamsToUpdate, relatedResourceIds }, ctx) => {
      const associateAttachmentTeamDto = {
        teamsToUpdateDto: teamsToUpdate,
        parentId,
        humanIds: files,
        relatedResourceIds,
      }
      const permissionJWT = await getUserPermissionsJWT(files, ctx)
      return ctx.loaders.attachment.associateAttachmentTeams(permissionJWT).load(associateAttachmentTeamDto, { humanId: parentId })
    },
  },

  SPARK_Attachment: {
    businessPartnersFull: (attachment, args, ctx) => {
      return attachment.businessPartners
        ? Promise.all(
          attachment.businessPartners.map(async bpID => {
            return !!bpID && (ctx.loaders.vmm.getByID.load(bpID))
          }),
        )
        : []
    },
    access: (attachment, args, ctx) =>
      ctx.loaders.accessControl.getPermissionsForResource.load(attachment.human_id || attachment.document_id),
    documentId: attachment => attachment.document_id || attachment.documentId,
    attachmentId: attachment => attachment.id,
    humanId: attachment => attachment.human_id || attachment.humanId,
    documentType: attachment =>
      attachment.document_type || attachment.documentType,
    documentSize: attachment =>
      attachment.document_size || attachment.documentSize,
    documentName: attachment =>
      attachment.document_name || attachment.name,
    productPacketProps: attachment => attachment.product_packet_props || attachment.productPacketProps,
    galleryDetails: attachment => attachment.gallery_details || attachment.galleryDetails,
    id: attachment => attachment.human_id || attachment.humanId || attachment.document_id || attachment.documentId,
    createdAt: attachment => attachment.created_at ? new Date(attachment.created_at) : new Date(attachment.createdAt),
    createdBy: (attachment, args, ctx) => {
      if(attachment.created_by || attachment.createdUser || attachment.createdBy) {
        return getUser(attachment.created_by || attachment.createdUser || attachment.createdBy, ctx)
      }
    },
    versionNumber: attachment => attachment.versionNumber || attachment.version_number,
    updatedAt: attachment => attachment.updated_at ? new Date(attachment.updated_at) : attachment.updateAt,
    updatedBy: (attachment, args, ctx) => {
      if(attachment.updated_by || attachment.updatedUser || attachment.updatedBy) {
        return getUser(
          attachment.updated_by || attachment.updatedUser || attachment.updatedBy, ctx,
        )
      }
    },
    canShareWithGallery: attachment => attachment.can_share_with_gallery === undefined ? attachment.canShareWithGallery : attachment.can_share_with_gallery,
    attachmentCanShowdog: attachment => attachment.can_share_with_gallery === undefined ? attachment.attachmentCanShowdog : attachment.can_share_with_gallery,
    canOpenInShowDog: attachment => attachment.can_open_in_showdog === undefined ? attachment.canOpenInShowDog : attachment.can_open_in_showdog,
    sharedWithGallery: attachment => attachment.can_share_with_gallery === undefined ? attachment.sharedWithGallery : attachment.can_share_with_gallery,
    modelFile: async (attachment, args, ctx) => {
      return get3DmodelFile(attachment, ctx)
    },
    tags: async (attachment, args, ctx) => {
      if (!attachment.tags) {
        return []
      }
      const tags = await ctx.loaders.tag.getTags.load({
        page: 0,
        size: 10000,
        ids: attachment.tags.join(','),
      })
      return tags.content || []
    },
    relatedResources: attachment =>
      attachment.related_resources || attachment.relatedResources,
    mediaType: attachment => attachment.media_type || attachment.mediaType,
    // False needs to be returned, not skipped in OR logic, so check for undefined instead
    finalVirtualFile: attachment =>
      attachment.final_virtual_file === undefined ? attachment.finalVirtualFile : attachment.final_virtual_file,
  },

  SPARK_GalleryFile: {
    canOpenInShowDog: file => {
      const openInShowDog = file.canOpenInShowDog === undefined ? file.can_open_in_show_dog : file.canOpenInShowDog
      return openInShowDog
    },
  },

  SPARK_3d_File: {
    fileId: file => file === undefined ? null : file.fileId,
    fileName: file => file === undefined ? null : file.fileName,
  },

  SPARK_ProductPacketProps: {
    partnerId: ProductPacketProps => ProductPacketProps.partner_id || ProductPacketProps.partnerId,
    critical: ProductPacketProps => ProductPacketProps.critical,
  },

  SPARK_GalleryDetails: {
    assetId: galleryDetails => galleryDetails.asset_id || galleryDetails.assetId,
    publishedAt: galleryDetails => galleryDetails.published_at || galleryDetails.publishedAt,
    fileValid: galleryDetails =>
    // False needs to be returned, not skipped in OR logic, so check for undefined instead
      galleryDetails.file_valid === undefined ? galleryDetails.fileValid : galleryDetails.file_valid,
    publishedBy: (galleryDetails, {}, ctx) =>
      (galleryDetails.published_by || galleryDetails.publishedBy) &&
      ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(
        galleryDetails.published_by || galleryDetails.publishedBy,
      ),
    fileTypes: (galleryDetails, {}, ctx) => (galleryDetails.asset_id || galleryDetails.assetId) &&
                  ctx.loaders.gallery.getAssetFiles.load(galleryDetails.asset_id || galleryDetails.assetId),
  },
}
