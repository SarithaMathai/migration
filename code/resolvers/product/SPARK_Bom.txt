const PRODUCT_HUB_MATERIAL_BOM_TYPE = { code: 1, description: 'Product' }

const searchMaterialById = async (variableName, detail, ctx) => {
  const id = get(detail, `${variableName}.id`, null)
  if (id) {
    const material = ctx.loaders.search.getMaterialByIds.load(id)
    return material
  }
}

const getMaterialHubResource = async (material, ctx) => {
  const matId = get(material, 'libraryResource.id') || null
  if (matId) {
    const permissionJWT = await getUserPermissionsJWT(matId, ctx)
    const materialHubData = await ctx.loaders.materialHub
      .getHubMaterial(permissionJWT)
      .load(matId)
    return materialHubData
  }
}

          queryPayload[`nestedSearchFilters[${index}].type`] = item.type
          queryPayload[`nestedSearchFilters[${index}].fieldPath`] = item.fieldPath

          queryPayload[`nestedSearchFilters[${index}].operator`] = item.operator

      parent,
      { comboId, partnerIds },
      ctx,
    ) => {
      const fabricSpecCombos = await SPARK_Combination.Query.searchFabricSpecCombos(null, { q: `parentComboId:${comboId}`, page: 0, size: 100 }, ctx)
      const suppliers = []
      if (fabricSpecCombos && fabricSpecCombos.content && fabricSpecCombos.content.length > 0) {
        const filteredFabricSpecCombos = fabricSpecCombos.content.filter(fabricSpecCombo => (fabricSpecCombo.fsId &&
          ((partnerIds && partnerIds.length > 0 && fabricSpecCombo && fabricSpecCombo.mvIds && fabricSpecCombo.mvIds.length === 1 && partnerIds.includes(fabricSpecCombo.mvIds[0])) ||
            !partnerIds || partnerIds.length === 0)))

          await ctx.loaders.vmm.getByID.load(fabricSpecCombo.fsId)
            .then(fs => {
              if (fs.bpName) {
                suppliers.push({
                  fabricSupplier: {
                    id: fabricSpecCombo.fsId,
                    name: fs.bpName,
                  },
                  fabricSpecCombo: fabricSpecCombo,
                })
              }
            })
        }))
      }
      return suppliers

    addBom: async (parent, { sparkBom }, ctx) => {
      const bom = await ctx.loaders.bom.addBom(sparkBom)

        throw new Error('Error creating bom \n' + JSON.stringify(bom))

    updateBom: async (parent, { sparkBom }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(sparkBom.humanId, ctx)
      if (
        sparkBom.workspaceContext &&
        (sparkBom.workspaceContext.addWorkspaces.length > 0 ||
          sparkBom.workspaceContext.removeWorkspaces.length > 0)
      ) {
        await workspaceAssociationHelper(
          ValidWorkspaceAssociationTypes.BOM,
          sparkBom.humanId,
          sparkBom.workspaceContext.addWorkspaces,
          sparkBom.workspaceContext.removeWorkspaces,
          ctx,
        )

      const bom = await ctx.loaders.bom.updateBom(permissionJWT, sparkBom)
      if (bom.validationErrors || bom.message) {
        throw new Error('Error updating bom \n' + JSON.stringify(bom))
      }
      if (sparkBom.businessPartners) {
        await ctx.loaders.bom.updatePermissions(permissionJWT).load(sparkBom)
      }
      return bom

      return ctx.loaders.fabric.getSpecificationByID.load(detail.fabricSpecId)
    },
    fabric: async (detail, args, ctx) => {
      const fabricRecordHumanId = detail.fabricRecordHumanId
      if (detail.type === 'combination' && fabricRecordHumanId) {
        const permissionJWT = await getUserPermissionsJWT(fabricRecordHumanId, ctx)
        return ctx.loaders.fabric.getByID(permissionJWT).load(fabricRecordHumanId)
      }
    },
    fabricId: async (detail) => {
      const fabricRecordHumanId = detail.fabricRecordHumanId
      if (detail.type === 'combination' && fabricRecordHumanId) {
        return fabricRecordHumanId
      }
    },
    relatedMaterials: async (detail, args, ctx) => {
      if (ctx.currentUser.internal) {
        if (detail.type === 'fabric_spec_combo') {
          const buckets = _.uniq(
            detail.intentLineIds.map(lineId => lineId.split('.')[0]),
          )
          if (buckets && buckets.length > 0) {
            const paddedBuckets = buckets.map(bucket => `"${bucket}"`)
            return ctx.loaders.search.getPage.load({
              q: `relatedAssetIds:(${paddedBuckets.join(' OR ')})`,
              page: 0,
              size: 300,
            })
          } else {
            return null
          }
        }
        return null
      } else {
        if (
          detail.type === 'fabric_spec_combo' &&
          detail.intentLineIds &&
          detail.intentLineIds.length > 0
        ) {
          const buckets = _.uniq(
            detail.intentLineIds.map(lineId => lineId.split('.')[0]),
          )
          const paddedBuckets = buckets.map(bucket => `"${bucket}"`)
          const permissionJWT = await getUserPermissionsJWT(
            detail.intentLineIds,
            ctx,
          )
          const proxyIds = detail.intentLineIds
          proxyIds.push(detail.parentComboId)
          return ctx.loaders.search
            .searchMaterialsByProxyIds(permissionJWT)
            .load({
              q: `relatedAssetIds:(${paddedBuckets.join(' OR ')})`,
              proxyIds: proxyIds,
              page: 0,
              size: 300,
            })
        } else {
          return null
        }
      }
    },
  },
}

export async function getMaterialWeight (material, ctx) {
  if (material && material.weight) {
    const MATERIAL_WEIGHT_IN_GRAM_CODE = 23
    const weightUomId = material && material.weightUomId ? material.weightUomId : MATERIAL_WEIGHT_IN_GRAM_CODE
    const uoms = await ctx.loaders.materialHub.getHubUnitsOfMeasure.load()
    const uom = uoms && uoms.find(uom => uom.code === weightUomId)
    return { value: material.weight, unitOfMeasure: uom }
  }
  return null
}

export async function getValueWithMaterialHubUom (value, unitOfMeasureCode, ctx) {
  if (value && unitOfMeasureCode) {
    const uoms = await ctx.loaders.materialHub.getHubUnitsOfMeasure.load()
    const uom = uoms && uoms.find(uom => uom.code === unitOfMeasureCode)
    return { value, unitOfMeasure: uom }
  }
  return null
}

export async function getCodedOptions (type, ctx) {
  const options = await ctx.loaders.materialHub.getHubCodedOptions.load(type) || []
  const datamap = new Map()
  options.forEach((option) => datamap.set(option.code, option.description))
  return datamap
}
