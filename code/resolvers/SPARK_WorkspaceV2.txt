export const getIsSingleProduct = (resourceList) => resourceList.length === 1 && resourceList[0].startsWith('PID')
export const getWorkspaceResourceString = workspaceId =>
  `Attr-${workspaceId}-resources`

export const filterSample = (samples, type) => {
  return (
    samples &&
    samples.filter(
      sample => sample.tracking &&
        sample.tracking.receivedDate &&
        type.includes(+sample.sampleType.code),
    ))
}

export const filterRoundSample = (samples, type) => {
  return (
    samples &&
    samples.filter(
      sample => sample.tracking &&
        (!sample.sampleRound || sample.sampleRound.roundNumber === 1) &&
        sample.tracking.receivedDate &&
        sample.sampleType.code === +type,
    ))
}

      const formattedParams = { ...searchParams }

        if (searchParams[key]) {
          formattedParams[key] = searchParams[key].join(',')
        }
      })
      return ctx.loaders.search.getWorkspacesPagedV2.load(
        _.omitBy(formattedParams, param => !param && param !== 0),
      )
    },
    getWorkspacesPagedV3: (parentValue, searchParams, ctx) => {
      const formattedParams = { ...searchParams }
      _.without(Object.keys(searchParams), 'size', 'page', 'q', 'designPartnerIds').forEach(key => {

          formattedParams[key] = searchParams[key].join(',')

      let suggestions
      let searchQuery
      if (q.toLowerCase().startsWith('smp-') === true) {
        searchQuery = { q, size, parentResourceType: 'PRODUCT', parentIds: `${associatedProducts}` }
        suggestions = await ctx.loaders.search.getSampleSuggestions.load(
          _.omitBy(searchQuery, param => !param && param !== 0),
        )
      } else {
        searchQuery = { q, size, workspaceId }
        suggestions = await ctx.loaders.search.getWorkspaceProductsSuggestions.load(
          _.omitBy(searchQuery, param => !param && param !== 0),

      const elasticResults = await ctx.loaders.product.getWorkspaceProducts.load({
        workspaceId: workspaceId,
        filter: filter || [],
        q: q || '',
        page: 0,
        size: 10000,
      })

        products: elasticResults.content.map(product => {
          return {
            id: product.humanId,
            sampleIds: product.samples.map(sample => {

    getWorkspacePackagingAttachments: async (parentValue, { workspaceId, bpId }, ctx) => {
      // fetching the variable here, so that you can more easily mock it in tests
      const { WORKSPACE_PACKAGING_TAG_ID } = process.env
      // WORKSPACE_PACKAGING_TAG_ID value must be set in env variables
      const packagingTagId = WORKSPACE_PACKAGING_TAG_ID ?? ''
      if (!packagingTagId) {
        throw new Error('Packaging tag ID is not defined please setup WORKSPACE_PACKAGING_TAG_ID variable')
      }
      const sortField = 'createdAt'
      const sortDirection = 'desc'
      const relatedIds = [workspaceId]
      let q = `tags:${packagingTagId}`
      if (bpId) {
        q += ` AND security.bps:${bpId}`
      }
      const result = await ctx.loaders.search.searchAttachments.load({
        q,
        relatedIds,
        page: 0,
        // hopefully we never have more than 500 packaging attachments in a single workspace
        size: 500,
        sort: sortField && sortDirection ? `${sortField},${sortDirection}` : undefined,
      })
      return result?.content || []
    },

  Mutation: {
    createWorkspaceV2: async (parent, { workspace, validateUnique }, ctx) => {
      workspace.workspaceType = workspace.workspaceType ? workspace.workspaceType : 103
      const response = await ctx.loaders.workspaceV2.createWorkspaceV2.load(workspace, {validateUnique: Boolean(validateUnique)})
      if(response.message?.startsWith('A workspace with the same criteria already exists')) {
            throw new GraphQLError('Workspace already exists')
      }
      return response
    },
    updateWorkspaceV2: async (parent, { workspace, validateUnique }, ctx) => {
      const permissionJWT = await getUserPermissionsJWT(workspace.id, ctx)
      workspace.workspaceType = workspace.workspaceType ? workspace.workspaceType : 103
      const response = await ctx.loaders.workspaceV2
        .updateWorkspaceV2(permissionJWT, Boolean(validateUnique))
        .load(workspace)
       if(response.message?.startsWith('A workspace with the same criteria already exists')) {
            throw new GraphQLError('Workspace already exists')
      }
      return response
    },
    changeWorkspace: async (parent, incomingDataPayload, ctx) => {
      const payload = {
        newWorkspaceId: incomingDataPayload.newWorkspaceId,
        oldWorkspaceId: incomingDataPayload.oldWorkspaceId,
        productHumanId: incomingDataPayload.productId,
        teams: incomingDataPayload.teams || [],
        removeWorkspaceOnly: incomingDataPayload.removeWorkspaceOnly || false,
      }

      const permissionJWT = await getUserPermissionsJWT(getWorkspaceResourceString(incomingDataPayload.newWorkspaceId), ctx)
      const changeWorkspaceRequest = await ctx.loaders.workspaceV2.changeWorkspace(permissionJWT)
      const changeWorkspaceResponse = await changeWorkspaceRequest.load(payload)
      if (changeWorkspaceResponse.validationErrors || changeWorkspaceResponse.message) {
        throw new Error('Error changing workspaces  \n' + JSON.stringify(changeWorkspaceResponse))
      }

      return changeWorkspaceResponse
    },

    addResourcesToWorkspaceV2: async (
      parent,
      { resourceList, workspaceId, setDates, designCycle },
      ctx,
    ) => {
      const resourcePermissionJWT = await getUserPermissionsJWT(
        getWorkspaceResourceString(workspaceId),
        ctx,
      )
      if (getIsSingleProduct(resourceList)) {
        const workspacePageObject = await Product.SPARK_Product.workspaces({ productId: resourceList[0] }, {}, ctx)
        if (!workspacePageObject) {
          return new Error(`Could not fetch workspaces for ${resourceList[0]}`)
        }
        const firstWorkspace = workspacePageObject.totalElements === 0
        // When adding workspace, have to initialize productWorkspaceAttributes by calling the product API
        ctx.loaders.product.updateViewToggle.load({
          productId: resourceList[0],
          workspaceId,
          hidden: false,
          firstWorkspace,
          ...(firstWorkspace
            ? { designCycle, setDates } : {}
          ),
        })
      }

      return ctx.loaders.workspaceV2
        .addResourcesToWorkspaceV2(resourcePermissionJWT)
        .load(resourceList, { workspaceId })
    },

    ) => {
      const resourcePermissionJWT = await getUserPermissionsJWT(
        getWorkspaceResourceString(workspaceId),

    workspaceBusinessPartnerActionsV2: async (
      parent,
      { actionType, values },
      ctx,
    ) => {
      switch (actionType) {
        case 'REMOVE_TEAM':
          { const resourcePermissionJWT = await getUserPermissionsJWT(
            getWorkspaceResourceString(values.workspaceId),
            ctx,
          )

            .removeTeamsFromWorkspaceV3(resourcePermissionJWT)

          })

            .removeWorkspaceBusinessPartnerV2(permissionJWT)
            .load({
              workspaceId: values.workspaceId,
              partnerId: values.partnerId,
            })
          break }
        case 'DROP_PARTNER':

          { const droppedFlag = false

            id: values.workspaceId, excludeTypeList: 'product',

              return !(attachment in resourcePermissions) || resourcePermissions[attachment].permissions.length === 0

              return !(discussionThread in resourcePermissions) || resourcePermissions[discussionThread].permissions.length === 0

              logger.debug('workspaceDropResponse', { workspacePromiseResponse: promiseValues[0] })
              logger.debug('discussionDropResponse', { discussionPromiseResponse: promiseValues[1] })
              if (discussionDropCallRequired && Array.isArray(promiseValues[1])) {
                discussionDropPassed = !promiseValues[1].some(droppedDiscussionResponse =>

              logger.debug('discussionDropPassed', { discussionCallStatus: discussionDropPassed })
              logger.debug('workspaceDropPassed', { workspaceCallStatus: workspaceDropPassed })
              const resourceDropCallsPassed = workspaceDropPassed && discussionDropPassed

              if (resourceDropCallsPassed) {
                const accessControlResponse = await ctx.loaders.accessControl.dropPartnerFromResources.load(toBeDroppedPermissionsMap)

                if (accessControlResponse) {
                  logger.debug('accessControlResponse.status', { responseStatus: accessControlResponse.status })
                  logger.debug('accessControlResponse comparison', { responseStatusCheck: accessControlResponse.status >= 400 })
                }

                /// /Only for debugging,will remove it
                if (accessControlResponse && accessControlResponse.status >= 400) {
                  // ACL drop call failed,so undropping the workspace and discussion call

                        workspaceId: values.workspaceId, partnerId: values.partnerId,

                // One or both of workspace or discussion drop calls have failed,
                // so we will undrop the successful call

                if (workspaceDropPassed) {
                  ctx.loaders.workspaceV2.unDropWorkspaceBusinessPartnerV2(userPermissionsCapabilityTokenJWT)
                    .load({}, { workspaceId: values.workspaceId, partnerId: values.partnerId })
                }
                if (discussionDropCallRequired && discussionDropPassed) {
                  ctx.loaders.discussion.unDropPartnerFromDiscussionIds(userPermissionsCapabilityTokenJWT).load({
                    ids: discussions, partnerIds: [parseInt(values.partnerId)],
                  })
                }
                // TODO
                // Promise.reject("Failed")
              }
            })

          let resourcePermissionsUndrop = {}

          const discussionsForUndrop = relationShipAsMapForUndrop && relationShipAsMapForUndrop.discussions ? relationShipAsMapForUndrop.discussions.map(({ id }) => id) : []

          const attachmentsForUndrop = relationShipAsMapForUndrop && relationShipAsMapForUndrop.attachments ? relationShipAsMapForUndrop.attachments.map(({ id }) => id) : []

            _.remove(attachmentsForUndrop, (attachment) =>

            )
            _.remove(discussionThreadsForUndrop, (discussionThread) =>
              !(discussionThread in resourcePermissionsUndrop) || resourcePermissionsUndrop[discussionThread].permissions.length === 0,

            permissionsMapForUndrop = getPermissionMapForBulkACLCall(resourcePermissionsUndrop, values.partnerId)
          }

          const userPermissionsCapabilityTokenJWTForUndrop = await getUserPermissionsJWT(
            [values.workspaceId, ...discussionsForUndrop, ...discussionThreadsForUndrop],

          Promise.all([
            ctx.loaders.workspaceV2.unDropWorkspaceBusinessPartnerV2(userPermissionsCapabilityTokenJWTForUndrop)

            discussionUnDropRequired && ctx.loaders.discussion.unDropPartnerFromDiscussionIds(userPermissionsCapabilityTokenJWTForUndrop).load({
              ids: discussionsForUndrop, partnerIds: [parseInt(values.partnerId)],

            .then(async promiseValues => {
              const workspaceUnDropStatus = !_.isEmpty(promiseValues[0].partnerId) && promiseValues[0].partnerId.includes(parseInt(values.partnerId))

              let discussionUnDropStatus = !discussionUnDropRequired

              if (discussionUnDropRequired && Array.isArray(promiseValues[1])) {
                discussionUnDropStatus = !promiseValues[1].some(unDroppedDiscussionResponse =>
                  _.isEmpty(unDroppedDiscussionResponse.partnerId) || !unDroppedDiscussionResponse.partnerId.includes(parseInt(values.partnerId)),
                )
              }
              const resourcesUnDropStatus = workspaceUnDropStatus && discussionUnDropStatus

              if (resourcesUnDropStatus) {
                const accessControlResponse = await ctx.loaders.accessControl.unDropPartnerFromResources.load(permissionsMapForUndrop)

                if (accessControlResponse && accessControlResponse.status >= 400) {
                  // ACL undrop call failed,so dropping the workspace and discussion call
                  Promise.all([
                    ctx.loaders.workspaceV2.dropWorkspaceBusinessPartnerV2(userPermissionsCapabilityTokenJWTForUndrop)
                      .load({}, {
                        workspaceId: values.workspaceId, partnerId: values.partnerId,

                    discussionUnDropRequired && ctx.loaders.discussion.dropPartnerFromDiscussionIds(userPermissionsCapabilityTokenJWTForUndrop).load({
                      ids: discussionsForUndrop, partnerIds: [parseInt(values.partnerId)],
                    })])

              } else {
                // One or both of workspace or discussion undrop calls have failed,
                // so we will drop the successful call
                if (workspaceUnDropStatus) {

                    .load({}, { workspaceId: values.workspaceId, partnerId: values.partnerId })

                if (discussionUnDropRequired && discussionUnDropStatus) {

          const dropUndropBody = { dropped, partnerType, workspaceId, partnerId }
          let [allResourcePermissions, toBePermissionsMap, allIds] = [{}, {}, []]

            id: values.workspaceId, excludeTypeList: 'product',

          const discussionThreadsList = relationShipMap && relationShipMap.discussionThreads ? relationShipMap.discussionThreads.map(({ id }) => id) : []

          const sampleList = relationShipMap && relationShipMap.sample ? relationShipMap.sample.map(({ id }) => id) : []

          const claimList = relationShipMap && relationShipMap.claim ? relationShipMap.claim.map(({ id }) => id) : []

          allIds = [...discussionList, ...attachmentList, ...discussionThreadsList, ...sampleList, workspaceId, ...claimList]

          if (allIds.length > 0) {
            allResourcePermissions = await filterResourcesByPartner(ctx, allIds, parseInt(partnerId), true, !dropped)
            _.remove(discussionList, (discussionId) =>
              !(discussionId in allResourcePermissions) || allResourcePermissions[discussionId].permissions.length === 0)
            _.remove(attachmentList, (attachment) =>
              !(attachment in allResourcePermissions) || allResourcePermissions[attachment].permissions.length === 0)
            _.remove(discussionThreadsList, (discussionThread) =>
              !(discussionThread in allResourcePermissions) || allResourcePermissions[discussionThread].permissions.length === 0)
            _.remove(sampleList, (samples) =>
              !(samples in allResourcePermissions) || allResourcePermissions[samples].permissions.length === 0)
            _.remove(claimList, (claim) => !(claim in allResourcePermissions) || allResourcePermissions[claim].permissions.length === 0)

            toBePermissionsMap = getPermissionMapForBulkACLCall(allResourcePermissions, partnerId)

          const capabilityToken = await getUserPermissionsJWT(
            [workspaceId, ...discussionList, ...discussionThreadsList, ...sampleList, ...claimList, SAMPLE_EVALUTION],
            ctx,

          const sampleCallNeeded = sampleList && sampleList.length > 0
          let sampleCall = false
          if ((partnerType !== BusinessPartnerRole.DESIGN_PARTNER.code) && sampleCallNeeded) {
            sampleCall = !dropped ? ctx.loaders.sampleV2.unDropSamples(capabilityToken).load(sampleList, {}) : ctx.loaders.sampleV2.dropSamples(capabilityToken).load(sampleList, {})
          }

          await Promise.all([
            ctx.loaders.workspaceV2.dropUndropWorkspaceBusinessPartnerV2(capabilityToken)
              .load(dropUndropBody, { workspaceId: workspaceId }), sampleCall]).then(async () => {
            dropped ? await ctx.loaders.accessControl.dropPartnerFromResources.load(toBePermissionsMap) : await ctx.loaders.accessControl.unDropPartnerFromResources.load(toBePermissionsMap)
            dropped && await UserProfileAttributes.Mutation.deleteAllUserProfileDataForAPartner(
              null, {
                partnerId: partnerId,
                resourceId: workspaceId,
                resourceType: 'workspace',
                discussionIds: discussionList || [],

          includeBranches: ['workspace', 'discussions', 'discussionThreads'],

        relations[0].nodes.map(({ id, type }) => {
          if (type === 'discussionThreads') {
            discussionThreadIds.push(id)
          } else if (type === 'discussions') {
            discussionIds.push(id)
          } else if (type === 'attachments_v3') {
            attachmentV3Ids.push(id)
          }
        })

        if (_.isEmpty(attachmentV3Ids)) {
          return []
        }

        const attachmentsV3JWT = (attachmentV3Ids.length === 0) ? '' : await ctx.loaders.accessControl.getUserAccessByPost.load(attachmentV3Ids)

        let attachments = await ctx.loaders.attachment.getAttachmentsV3(attachmentsV3JWT).load(attachmentV3Ids)
        const { resourcePermissions: currentUserPermissions } = await ctx.loaders.accessControl.getUserAccessUnencoded.load(

        let discussionThreads = []

        if (discussionIds.length) {
          discussions = await getDiscussionsBatch(discussionIds, CHUNK_SIZE, ctx)
        }

        if (discussionThreadIds.length) {
          discussionThreads = await getDiscussionThreadsBatch(discussionThreadIds, CHUNK_SIZE, ctx)
        }

        return _.orderBy(attachments, ['resource.type', 'created_at'], ['desc', 'desc'])
          .map(attachment => {
            const relatedDiscussion = _.find(discussions, { discussionId: attachment.resource.id })
            const relatedDiscussionThread = _.find(discussionThreads, { discussionThreadId: attachment.resource.id })
            let relatedResource = {}

            if (relatedDiscussionThread) {
              relatedResource = _.find(discussions, { discussionId: relatedDiscussionThread.discussionId })
            } else if (relatedDiscussion) {
              relatedResource = relatedDiscussion
            }
            return {
              attachment,
              linkedResourceId: relatedResource.discussionId,
              resourceTitle: relatedResource.subject,
              isResourceDraft: relatedDiscussionThread ? relatedDiscussionThread.draft : relatedResource.draft || false,
              currentUserPermissions: _.find(currentUserPermissions, { resourceId: attachment.document_id }),
            }
          })
          .filter(({ attachment, linkedResourceId, isResourceDraft }) => {
            if (attachment.resource.type === 'discussion' && (!linkedResourceId || isResourceDraft)) {
              // TODO: ACL should bbe doing this
              // User should not have access to this draft attachment
              return false
            }
            return true
          })
      } else {
        return []
      }

      let productFirstRoundSample = 0
      if (_.get(workspace, 'resources.mappings.product', []).length === 0) {

          virtualSampleCount = +filterSample(samples.content, [VIRTUAL_SAMPLE_TYPE]).length || 0
          workspaceSamplesCount = +filterSample(samples.content, [PRODUCT_SAMPLE_TYPE]).length + virtualSampleCount

      if (_.get(workspace, 'resources.mappings.product', []).length === 0) {
        return 0
      } else {
        const filterParams = {
          resourceType: 'workspaces',
          resourceId: info.variableValues.id,
          filter: (args.filter || info.variableValues.filter) || [],
          page: args.page,
          size: args.size,

        args.q && Object.assign(filterParams, { q: args.q })

        const filteredProductsWithSummary = await ctx.loaders.search.getFilteredProductsWithSummary.load({
          ...filterParams,
        })

        const productData = await ctx.loaders.product.getPage.load({
          products: filteredProductsWithSummary?.products || [],
          page: args.page,
          size: args.size,
          additionalParams: '',
        })

        if (productData.content && productData.content.length > 0) {
          const productIds = productData.content.map(product => product.productId)
          const discussionResponse = await ctx.loaders.discussion.getDiscussionsCount.load(
            {
              resourceType: 'product',
              resourceId: productIds,
            },
          )
          const sampleResponse = await ctx.loaders.search.getSampleCount.load(
            productIds,
          )
          const samples = await ctx.loaders.search.searchSamplesByParentId.load({
            page: 0,
            query: '*',
            parentIds: productIds,
            size: 10000,
          })
          let discussionSampleResponse = {}
          if (samples.content && samples.content.length > 0) {
            discussionSampleResponse = await ctx.loaders.discussion.getDiscussionsCount.load(
              {
                resourceType: 'sample',
                resourceId: samples.content.map(sample => sample.id),
              },
            )
            discussionSampleResponse.discussionsCount.map(sampleDiscussion => {
              // For each sample discussion (if count is 0 this is false) find the whole sample
              if (sampleDiscussion.count) {
                const sample = samples.content.find(
                  sample => sample.id === sampleDiscussion.resourceId,
                )
                // With that sample find the product Discussion count entry that matches its parent (the product)
                if (sample) {
                  const discussionCountForProduct = discussionResponse.discussionsCount.find(
                    discussionCount =>
                      discussionCount.resourceId === sample.parentId,
                  )
                  // If you found it increment it by 1
                  if (discussionCountForProduct) {
                    discussionCountForProduct.count += 1
                  }
                }
              }
            })
          }
          return {
            productsCount: productData.totalElements,
            productsDiscussionsCount: discussionResponse.discussionsCount,
            productsSamplesCount: sampleResponse,
            nonEvaluatedRecievedSamplesCount: filteredProductsWithSummary?.receivedNotEvaluatedByWorkspaceCount || 0,
            workspaceProductDashboard: filteredProductsWithSummary?.workspaceProductDashboard || null,
          }
        } else {
          return {
            productsCount: 0,
            productsDiscussionsCount: [],
            productsSamplesCount: [],
            nonEvaluatedRecievedSamplesCount: 0,
          }
        }
      }
    },
