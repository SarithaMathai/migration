  Mutation: {
    addDiscussionV2: async (parent, { sparkDiscussion, attachments }, ctx) => {
      let relationshipNodes = []
      const resourceType = sparkDiscussion.resourceType
      const resourceId = sparkDiscussion.resourceId
      relationshipNodes = [
        {
          newRelationshipRootDTO: {
            root: resourceId,
            rootResourceType: resourceType,
          },
          newRelationshipNodeDTO: [
            {
              node: sparkDiscussion.discussionId,
              nodeResourceType: 'discussions',
            },
          ],
        },
      ]
      if (!_.isEmpty(attachments)) {
        relationshipNodes.push({
          newRelationshipRootDTO: {
            root: sparkDiscussion.discussionId,
            rootResourceType: 'discussions',
          },
          newRelationshipNodeDTO: attachments.map(file => {
            return {
              node: file.documentId,
              nodeResourceType: 'attachments',
            }
          }),
        })
      }
      if (!_.isEmpty(relationshipNodes)) {
        const relationship = await addBulkRelationShip(ctx, relationshipNodes)
        if (relationship.status && relationship.status >= 400) {
          return Promise.reject(relationship.error)
        }
      }
      // Create bulk attachment should happen only after creating relationships.
      if (!_.isEmpty(attachments)) {
        const permissionJWT = await getUserPermissionsJWT(
          attachments.map(attachment => attachment.documentId),
          ctx,
        )
        await ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load({ attachments: attachments })
      }

      sparkDiscussion = { ...sparkDiscussion, enableAcl: true }
      const discussion = await ctx.loaders.discussion.createV2.load(
        sparkDiscussion,
      )
      return discussion
    },

    addDiscussionReplyV2: async (
      parent,
      { sparkDiscussionReply, attachments, isAttachmentsV3 },
      ctx,
    ) => {
      const aclPermissionJWT = await getUserPermissionsJWT(
        sparkDiscussionReply.discussionId,
        ctx,
      )
      const relationshipNodes = [
        {
          newRelationshipRootDTO: {
            root: sparkDiscussionReply.discussionId,
            rootResourceType: 'discussions',
          },
          newRelationshipNodeDTO: [
            {
              node: sparkDiscussionReply.discussionThreadId,
              nodeResourceType: 'discussionThreads',
            },
          ],
        },
      ]

      if (!_.isEmpty(attachments)) {
        relationshipNodes.push({
          newRelationshipRootDTO: {
            root: sparkDiscussionReply.discussionThreadId,
            rootResourceType: 'discussionThreads',
          },
          newRelationshipNodeDTO: attachments.map(file => {
            return {
              node: file.documentId,
              nodeResourceType: isAttachmentsV3 ? 'attachments_v3' : 'attachments',
            }
          }),
        })
      }
      if (!_.isEmpty(relationshipNodes)) {
        const relationship = await addBulkRelationShip(ctx, relationshipNodes)
        if (relationship.status && relationship.status >= 400) {
          return Promise.reject(relationship.error)
        }
      }
      if (!_.isEmpty(attachments)) {
        const permissionJWT = await getUserPermissionsJWT(
          attachments.map(attachment => attachment.documentId),
          ctx,
        )
        attachments.forEach(attachment => { attachment.draft = sparkDiscussionReply.draft })
        const attachmentUpdateResult = await ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load({ attachments: attachments })
        if (attachmentUpdateResult.status_code && attachmentUpdateResult.status_code >= 400) {
          return Promise.reject('Error updating attachments for discussion: ' + attachmentUpdateResult.message)
        }
      }
      sparkDiscussionReply = { ...sparkDiscussionReply, enableAcl: true }
      return ctx.loaders.discussion
        .createReplyV2(aclPermissionJWT)
        .load(sparkDiscussionReply, sparkDiscussionReply.discussionId)
    },

    updateDiscussionV2: async (parent, { id, sparkDiscussion }, ctx) => {
      const aclPermissionJWT = await getUserPermissionsJWT(id, ctx)
      const deleteNodes = []
      sparkDiscussion.participants.partnerId = sparkDiscussion.participants.partnerId.filter(Number)

      const deleteAttachmentList = []
      if (!_.isEmpty(sparkDiscussion.removedFiles)) {
        const attachmentPermissionJWT = await getUserPermissionsJWT(
          sparkDiscussion.removedFiles,
          ctx,
        )
        _.forEach(sparkDiscussion.removedFiles, document => {
          deleteAttachmentList.push({
            documentId: document,
          })
          deleteNodes.push({
            id: document,
          })
        })
        const bulkAttachmentDeleteDTO = { attachments: deleteAttachmentList }
        await ctx.loaders.attachment
          .archiveAttachmentBulkV2(attachmentPermissionJWT)
          .load(bulkAttachmentDeleteDTO)
        await removeRelationship({
          ctx,
          rootId: id,
          nodes: deleteNodes,
        })
      }
      let deleteParticipants = {}
      if (!_.isEmpty(sparkDiscussion.deletedTeams)) {
        const deletedTeamsList = []
        sparkDiscussion.deletedTeams.map(team => {
          deletedTeamsList.push({ teamId: team })
        })
        deleteParticipants.teams = deletedTeamsList
      }
      if (!_.isEmpty(sparkDiscussion.deletedUsers)) {
        deleteParticipants.users = sparkDiscussion.deletedUsers
      }
      if (!_.isEmpty(sparkDiscussion.deletedPartnersId)) {
        deleteParticipants.partners = sparkDiscussion.deletedPartnersId
      }

      if (!_.isEmpty(deleteParticipants)) {
        deleteParticipants = { ...deleteParticipants, enableAcl: true }
        await ctx.loaders.discussion
          .deleteParticipantsV2(aclPermissionJWT)
          .load(deleteParticipants, { id })
      }
      if (!_.isEmpty(sparkDiscussion.attachments)) {
        const relationshipNodes = [
          {
            newRelationshipRootDTO: {
              root: sparkDiscussion.discussion.discussionId,
              rootResourceType: 'discussions',
            },
            newRelationshipNodeDTO: sparkDiscussion.attachments.map(file => {
              return {
                node: file.documentId,
                nodeResourceType: 'attachments',
              }
            }),
          },
        ]
        const relationship = await addBulkRelationShip(ctx, relationshipNodes)
        if (relationship.status && relationship.status >= 400) {
          return Promise.reject(relationship.error)
        }
        const permissionJWT = await getUserPermissionsJWT(
          sparkDiscussion.attachments.map(attachment => attachment.documentId),
          ctx,
        )
        await ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load(
          { attachments: sparkDiscussion.attachments },
        )
      }
      if (!_.isEmpty(sparkDiscussion.participants)) {
        sparkDiscussion.participants = { ...sparkDiscussion.participants, enableAcl: true }
        await ctx.loaders.discussion
          .updateParticipantsV2(aclPermissionJWT)
          .load(sparkDiscussion.participants, { id })
      }
      sparkDiscussion.discussion = { ...sparkDiscussion.discussion, enableAcl: true }
      const discussionList = await ctx.loaders.discussion
        .updateV2(aclPermissionJWT)
        .load(sparkDiscussion.discussion)
      return discussionList
    },

    deleteParticipantV2: async (
      parent,
      { id, teamId, userId, partnerId },
      ctx,
    ) => {
      const aclPermissionJWT = await getUserPermissionsJWT(id, ctx)
      let deleteParticipants = {}
      if (!teamId || teamId === 'users') {
        deleteParticipants.users = [userId]
        if (partnerId) {
          deleteParticipants.partners = [partnerId]
        }
      } else {
        const deletedTeamsList = []
        if (userId) {
          deletedTeamsList.push({ teamId: teamId, users: [userId] })
        } else {
          deletedTeamsList.push({ teamId: teamId })
        }
        deleteParticipants.teams = deletedTeamsList
        if (partnerId) {
          deleteParticipants.partners = [partnerId]
        }
      }
      deleteParticipants = { ...deleteParticipants, enableAcl: true }
      const removedParticipant = await ctx.loaders.discussion
        .deleteParticipantsV2(aclPermissionJWT)
        .load(deleteParticipants, { id })
      return removedParticipant
    },

    updateParticipantsV2: async (parent, { id, participants }, ctx) => {
      participants.partnerId = participants.partnerId.filter(Number)

      await ctx.loaders.discussion

        .load(participants, {
          id,

    deleteDiscussionPartnersV2: async (parent, { id, partnerIds }, ctx) => {

      deleteParticipants = { ...deleteParticipants, enableAcl: true }

        .deleteParticipantsV2(aclPermissionJWT)
        .load({ deleteParticipants, id })
    },

    updateDiscussionReplyV2: async (
      parent,
      { sparkDiscussionReply, attachments, isAttachmentsV3 },
      ctx,
    ) => {
      const aclPermissionJWT = await getUserPermissionsJWT(
        sparkDiscussionReply.discussionThreadId,
        ctx,
      )

      if (!_.isEmpty(attachments)) {
        const relationshipNodes = [
          {
            newRelationshipRootDTO: {
              root: sparkDiscussionReply.discussionThreadId,
              rootResourceType: 'discussionThreads',
            },
            newRelationshipNodeDTO: attachments.map(file => {
              return {
                node: file.documentId,
                nodeResourceType: isAttachmentsV3 ? 'attachments_v3' : 'attachments',
              }
            }),
          },
        ]

        const relationship = await addBulkRelationShip(ctx, relationshipNodes)
        if (relationship.status && relationship.status >= 400) {
          return Promise.reject(relationship.error)
        }
        const permissionJWT = await getUserPermissionsJWT(
          attachments.map(attachment => attachment.documentId),
          ctx,
        )
        await ctx.loaders.attachment.bulkUpdateAttributes(permissionJWT).load({ attachments: attachments })
        if (!ctx.currentUser.internal && sparkDiscussionReply.sampleFormatCode === VIRTUAL_SAMPLE_FORMAT && !sparkDiscussionReply.draft) {
          await handle3dFileSampleDiscussion(sparkDiscussionReply, attachments.map(attachment => attachment.documentId), ctx)
        }
      }

      sparkDiscussionReply = { ...sparkDiscussionReply, enableAcl: true }
      const reply = await ctx.loaders.discussion
        .updateReplyV2(aclPermissionJWT)
        .load(sparkDiscussionReply)
      if (reply.validationErrors || reply.message) {
        throw new Error('Error updating reply  \n' + JSON.stringify(reply))
      }
      return reply
    },

    deleteDiscussionV2: async (parent, { resourceId, id }, ctx) => {
      const aclPermissionJWT = await getUserPermissionsJWT(id, ctx)
      const attachmentData = await ctx.loaders.search.searchAttachments.load({
        relatedIds: [id],
      })
      const attachments = attachmentData ? attachmentData.content : []
      if (attachments && attachments.length) {
        const humanIds = attachments.map(attachment => attachment.id)
        const permissions = await getUserPermissionsJWT(humanIds, ctx)
        const archiveAttachmentPromises = humanIds?.map(it => ctx.loaders.attachment.archiveAttachmentV3(permissions).load(it))
        await Promise.all(archiveAttachmentPromises)
      }
      await removeRelationship({
        ctx,
        rootId: resourceId,
        nodes: [{ id }],
      })
      ctx.loaders.discussion.deleteV2(aclPermissionJWT).load(id)
    },

    deleteDiscussionReplyV2: async (parent, { id, replyId }, ctx) => {
      const aclPermissionJWT = await getUserPermissionsJWT(replyId, ctx)
      const attachmentData = await ctx.loaders.search.searchAttachments.load({
        relatedIds: [replyId],
      })
      const attachments = attachmentData ? attachmentData.content : []
      if (attachments && attachments.length) {

        const archiveAttachmentPromises = humanIds?.map(it => ctx.loaders.attachment.archiveAttachmentV3(permissions).load(it))

        ctx,
        rootId: id,
        nodes: [{ id: replyId }],
      })
      ctx.loaders.discussion
        .deleteReplyV2(aclPermissionJWT)
        .load({ id, replyId })
    },

  SPARK_FullDiscussion: {
    discussions: discussion => discussion,
    attachmentsV2: async (discussion, args, ctx) => {
      const resourceId = discussion.discussion_id || discussion.discussionId
      let attachments = []
      // attachment elastic call gives all the attachments associate with resourceId
      const attachmentsFromElastic = await ctx.loaders.search.searchAttachmentsByRelatedResource.load(resourceId)
      const attachmentsOnlyForResource = attachmentsFromElastic?.filter(attachment => attachment.parentResource === resourceId) || []
      if (!attachmentsOnlyForResource || attachmentsOnlyForResource.length < 1) return []
      attachments = attachmentsOnlyForResource
      return attachments

  SPARK_Discussion: {
    id: discussion => discussion.discussionId,
    threadId: discussion => discussion.discussionThreadId,
    authorName: discussion => discussion.authorName,
    repliesCount: discussion => discussion.repliesCount || 0,
    draftReplyCount: (discussion, args, ctx) => {
      const currentUser = ctx.currentUser
      if (currentUser.internal) {
        return discussion.internalDraftReplies || 0
      }
      const partnerId = currentUser.partnerId
      const getPartnerReplyById = _.find(discussion.partnerReplySummary, { 'businessPartnerId': partnerId }) || {}
      return getPartnerReplyById.draftCount || 0
    },
    createdTS: discussion => discussion.createdTs || discussion.createdAt,
    updateTS: discussion => discussion.updateTs || discussion.updatedAt,
    comment: discussion => discussion.comment,
    editable: discussion => discussion.editable,
    draft: discussion => discussion.draft,
    edited: discussion => discussion.edited,
    discussionTitle: discussion => discussion.subject,
    partnerId: async (discussion, args, ctx) => {
      let partnerId = []
      const existingBPGroup = getACLGroupFromPermissions(
        await getACLPermissionsForResource(ctx, discussion.discussionId),
        'bps',
      )
      if (!_.isEmpty(existingBPGroup)) {
        partnerId = existingBPGroup.grantees
      } else {
        partnerId = discussion.partnerId
      }
      return partnerId
    },
    isSampleDiscussion: discussion => {
      return discussion.parentId ? discussion.parentId.startsWith('SMP-') : discussion.sampleDiscussion
    },
    designPartnerId: async (discussion, args, ctx) => {
      let partnerId = await getDesignPartner(discussion.discussionId, ctx)
      // TODO - MVs are not getting DP permission from the ACL call, so reading design partner id from discussion service if acl is giving empty value
      return partnerId && partnerId.length ? partnerId : discussion.designPartnerId
    },
    updatedUser: (discussion, args, ctx) =>
      ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(discussion.updatedUser || discussion.updatedBy),
    createdUser: (discussion, args, ctx) =>
      ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(discussion.createdUser || discussion.createdBy),
    lastReplyBy: (discussion, args, ctx) =>
      discussion.lastReplyBy &&
      ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(discussion.lastReplyBy),
    participant: (discussion, args, ctx) =>
      discussion.participant.map(user => {
        return ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(user)
      }),
    participantDetails: async (discussion, args, ctx) => {
      if (!ctx.featureConfig.coreDiscussionFeatureFlag) {
        if (discussion.participantDetails) {
          return discussion.participantDetails
        } else {
          let resourceId = getDiscussionResourceId(discussion, ctx)
          return getParticipantsByUserGroup(resourceId, ctx)
        }
      } else {
        let resourceId = getDiscussionResourceId(discussion, ctx)
        return getParticipantsByUserGroup(resourceId, ctx)
      }
    },
    resources: discussion => {
      if (discussion.resources) {
        return discussion.resources
      } else {
        const parentResourceType = discussion.parentResourceType ? discussion.parentResourceType : getParentResourceType(discussion.parentId)
        let resources = { mappings: { product: null, sample: null, workspace: null } }
        resources.mappings[parentResourceType] = [discussion.parentId]
        return resources
      }
    },
    participants: discussion => discussion.participantDetails,
    subject: discussion => discussion.subject,
    replies: async (discussion, args, ctx) => {
      if (discussion.discussionReplies) {
        return discussion.discussionReplies
      } else {
        return ctx.loaders.discussion.getThreads.load({
          id: discussion.discussionId,
          page: 0,
          size: 20,
        })
      }
    },
    repliesV2: async (discussion, args, ctx) => getDiscussionRepliesV2(discussion.discussionId, ctx) || [],
    resource: (discussion, args, ctx) => getDiscussionResource(discussion, ctx),
    businessPartner: async (discussion, args, ctx) => {
      let partnerId = []
      const bPGroup = getACLGroupFromPermissions(
        await getACLPermissionsForResource(ctx, discussion.discussionId),
        'bps',
      )
      if (!_.isEmpty(bPGroup)) {
        partnerId = bPGroup.grantees
      } else {
        partnerId = discussion.partnerId
      }
      return _.isEmpty(partnerId)
        ? []
        : ctx.loaders.vmm.getByID.loadMany(partnerId)
    },
    droppedPartners: async (discussion, args, ctx) => {
      let droppedPartnersWithType = []
      const droppedPartners = getDroppedPartnersFromACL(await getACLPermissionsForResource(ctx, discussion.discussionId))
      const droppedPartnerIds = droppedPartners.map(partner => partner.id).filter(Boolean)
      const bps = _.isEmpty(droppedPartnerIds) && droppedPartnerIds.length
        ? []
        : await ctx.loaders.vmm.getByID.loadMany(droppedPartnerIds)

      bps.forEach(bp => {
        const productPartner = droppedPartners.filter(partner => parseInt(partner.id) === parseInt(bp.bpId))

        if (productPartner[0]) {
          droppedPartnersWithType.push({
            ...bp,
            associatedPartnerType: productPartner[0].type || 8,
          })
        }
      })
      return droppedPartnersWithType
    },
    designPartners: async (discussion, args, ctx) => {
      let partnerId = await getDesignPartner(discussion.discussionId, ctx)
      // TODO - MVs are not getting DP permission from the ACL call, so reading design partner id from discussion service if acl is giving empty value
      const designPartnerId = partnerId && partnerId.length ? partnerId : discussion.designPartnerId
      if (!designPartnerId) {
        return []
      }
      const vmm = await ctx.loaders.vmm.getByID.load(designPartnerId)
      vmm.associatedPartnerType = 5
      return [vmm]
    },
    tags: async (discussion, args, ctx) => {
      let tagIds = _.get(discussion, 'tags') || []
      let tags = []
      if (tagIds.length > 0) {
        tags = getTagsBatched(ctx, discussion.tags)
      }
      return tags
    },
    readByUserList: discussion => discussion.readByUserList,
    associations: discussion => discussion.associations,

  },

  SPARK_Discussion_Participants: {
    teams: participants => participants.teams,
    users: participants => participants.users,
  },

  SPARK_Discussion_Team: {
    teamId: dt => dt.teamId,
    edited: dt => dt.edited,
    teamName: dt => dt.teamName,
    businessPartner: async (team, args, ctx) => {
      if (team.businessPartnerId === 0 || team.businessPartnerId === null) {
        return { bpId: 0, bpName: 'Target' }
      } else {
        const businessPartner = await ctx.loaders.vmm.getByID.load(
          team.businessPartnerId,
        )
        return businessPartner
      }
    },
    usersInfo: dt => dt.usersInfo,
  },

  SPARK_Discussion_Participant: {
    roles: member => member.roles,
    userId: member => member.userId,
    userDetails: async (member, args, ctx) =>
      ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(member.userId),
  },

  SPARK_Resource: {
    __resolveType: (resource) => {
      if (resource.workspaceHumanId) {
        return 'SPARK_WorkspaceV2'
      } else if (resource.productId) {
        return 'SPARK_Product'
      } else if (resource.humanId) {
        const humanIdParts = resource.humanId.split('-')
        const humanIdPrefix = (humanIdParts && humanIdParts.length > 0) ? humanIdParts[0] : ''
        if (humanIdPrefix.includes('PAL')) {
          return 'SPARK_Palette'
        } else if (humanIdPrefix.includes('ART')) {
          return 'SPARK_Artwork'
        } else if (humanIdPrefix.includes('TRM')) {
          return 'SPARK_Trim'
        } else if (humanIdPrefix.includes('WSH')) {
          return 'SPARK_Wash'
        } else if (humanIdPrefix.includes('FAB')) {
          return 'SPARK_Fabric'
        } else if (humanIdPrefix.includes('ARCCLR')) {
          return 'SPARK_ColorArchroma'
        } else if (humanIdPrefix.includes('CLR')) {
          return 'SPARK_Color'
        } else if (humanIdPrefix.includes('CBO')) {
          return 'SPARK_Combination'
        } else if (humanIdPrefix.includes('MAT')) {
          return 'SPARK_HubMaterial'
        } else if (humanIdPrefix.includes('BOM')) {
          return 'SPARK_Bom'
        } else if (humanIdPrefix.includes('PDTL')) {
          return 'SPARK_ProductDetails'
        } else if (humanIdPrefix.includes('MST')) {
          return 'SPARK_Measurements'
        } else if (humanIdPrefix.includes('CLM')) {
          return 'SPARK_Claims'
        } else if (humanIdPrefix.includes('PKG')) {
          return 'SPARK_Packaging'
        }
      } else {
        return 'SPARK_SampleV2'
      }
    },
  },

  SPARK_DiscussionReply: {
    id: reply => reply.discussionThreadId || reply.discussionId,
    discussionId: reply => reply.discussionId,
    comment: reply => reply.comment,
    discussionTitle: reply => reply.discussionTitle,
    draft: reply => reply.draft,
    edited: reply => reply.edited,
    createdTS: reply => reply.createdTs || reply.createdTS,
    updateTS: reply => reply.updatedTs || reply.updateTS,
    discussionHead: async (reply, args, ctx) => {
      return ctx.loaders.discussion.getByIDV2.load(reply.discussionId)
    },
    attachments: async (reply, args, ctx) => {
      const replyAttachments = await ctx.loaders.attachment.search.load({
        resourceId: reply.discussionThreadId,
        resourceType: 'discussion',
      })
      return replyAttachments
    },
    attachmentsV2: async (reply, {}, ctx) => {
      const attachments = await ctx.loaders.search.searchAttachmentsByRelatedResource.load(reply.discussionThreadId)
      if (!attachments || attachments.length < 1) return []
      return attachments
    },
    createdUser: (reply, args, ctx) => {
      if (reply.createdUser) {
        return ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(reply.createdUser)
      }
    },
    updatedUser: (reply, args, ctx) => {
      if (reply.updatedUser) {
        return ctx.loaders.userAttributes.getUserByIDOrNullIfNotFound.load(reply.updatedUser)
      }
    },
    readByUserList: reply => reply.readByUserList,
    associations: reply => reply.associations,
  },

  SPARK_NotificationStatus: {
    sentTime: status => new Date(status.sentTime),
  },
}

export async function getDiscussion (resourceId, ctx) {
  let discussions = []
  const children = await ctx.loaders.relationship.searchByIds.load({
    id: resourceId,
    maxDepth: 0,
  })

  if (children.length && !_.isEmpty(children[0].nodes)) {
    const discussionIds = children[0].nodes.filter(
      child => child.type === 'discussions',
    )

    if (discussionIds.length > 0) {
      const aclPermissionJWT = await ctx.loaders.accessControl.getUserAccessByPost.load(
        discussionIds.map(node => node.id),
      )
      discussions = await ctx.loaders.discussion
        .getDiscussionsByPost(aclPermissionJWT)
        .load(discussionIds.map(node => node.id))
    }
  }
  return discussions
}
