export default {
  Query: {
    getDivisions: (parent, args, ctx) =>
      ctx.loaders.ig.division.getPage.load(args),

    getDivision: (parentValue, { id }, ctx) =>
      ctx.loaders.ig.division.getByID.load(id),
  },

  IG_Division: {
    id: division => division.division_id,
    name: division => division.division_name,
    displayText: division =>
      `${division.division_id}: ${division.division_name}`,
  },
}
