export const orderProductAttachments = attachmentsWithMetaData => {
  const orderByResourceType = ({ attachment }) =>
    ['product', 'discussion', 'sample'].indexOf(attachment.resource.type)
  const orderByCreatedDate = ({ attachment }) =>
    new Date(attachment.created_at)
  return _.orderBy(
    attachmentsWithMetaData,
    [orderByResourceType, orderByCreatedDate],
    ['asc', 'desc'],
  )
}

const copyProductToProduct = async (copyProduct, ctx) => {
  if (!_.isEmpty(copyProduct)) {
    const userPermissionsCapabilityTokenJWT = await getUserPermissionsJWT(
      copyProduct.targetProductId,
      ctx,
    )
    const copyProductResponse = await ctx.loaders.product
      .copyProduct(userPermissionsCapabilityTokenJWT)
      .load(copyProduct)
    return copyProductResponse
  }
}

export const getTechPackResourceCountMap = async (
  productId,
  partnerId,
  workspaceContext,
  parentProductId,
  ctx,
) => {
  let outputMap = {}
  const inputMap = {
    attachments: [0, 1, 2, 3],
    attachments_v3: [0, 1, 2, 3],
  }
  let enhancedAttachments = []
  let parentDiscussionAttachments = []
  const partnerIdInt = parseInt(partnerId)
  // NOTE: 'product' can not be added to includeBranches without absorbing all attachments from ABO linked Products.
  const includeBranches = [
    'sample',
    'discussions',
    'discussionThreads',
    'bill_of_materials',
    'packaging_bom',
    'claim',
    'measurement_set',
    'construction_set',
    'product_watchlist',
  ]
  const includeNodeTypes = [
    'discussions',
    'discussionThreads',
    'attachments_v3',
  ]
  const commonLoaderOutputMaps = await getFilteredResourceAndLevelWiseChildrenBasedOnPartnerACLPermission(
    ctx,
    inputMap,
    productId,
    partnerIdInt,
    includeBranches,
    includeNodeTypes,
    true,
  )
  const parentProductOutputMaps = parentProductId
    ? await getFilteredResourceAndLevelWiseChildrenBasedOnPartnerACLPermission(
      ctx,
      inputMap,
      parentProductId,
      partnerIdInt,
      includeBranches,
      includeNodeTypes,
      true,
    )
    : null

  Object.keys(inputMap).forEach(key => {
    inputMap[key].map(mapedValues => {
      if (commonLoaderOutputMaps[key][mapedValues] === undefined) {
        commonLoaderOutputMaps[key][mapedValues] = []
      }
      if (
        parentProductOutputMaps &&
        parentProductOutputMaps[key][mapedValues] === undefined
      ) {
        parentProductOutputMaps[key][mapedValues] = []
      }
    })
  })

  if (parentProductOutputMaps) {
    parentDiscussionAttachments = [
      ...Object.values(parentProductOutputMaps['attachments_v3'][1]),
      ...Object.values(parentProductOutputMaps['attachments_v3'][2]),
      ...Object.values(parentProductOutputMaps['attachments_v3'][3]),
    ]
  }

  const commonLoaderAttachments = commonLoaderOutputMaps['attachments']
  const commonLoaderAttachments_v3 = commonLoaderOutputMaps['attachments_v3']

  let [productAttachments = [], ...discussionAttachments] = Object.values(
    _.mergeWith(
      commonLoaderAttachments,
      commonLoaderAttachments_v3,
      (attachments, attachments_v3) => attachments.concat(attachments_v3),
    ),
  )

  discussionAttachments = [
    ...parentDiscussionAttachments,
    ...Object.values(commonLoaderOutputMaps['attachments_v3'][1]),
    ...Object.values(commonLoaderOutputMaps['attachments_v3'][2]),
    ...Object.values(commonLoaderOutputMaps['attachments_v3'][3]),
  ]
  let allAttachmentIds = [...productAttachments, ...discussionAttachments]

  if (allAttachmentIds.length > 0) {
    const permissionJWT = await getUserPermissionsJWT(allAttachmentIds, ctx)
    enhancedAttachments = await ctx.loaders.attachment
      .getAttachmentsV3(permissionJWT)
      .load(allAttachmentIds)
  }

  let isProductPacketAttachments = attachmentId =>
    enhancedAttachments.some(
      enhancedAttachment =>
        (enhancedAttachment?.document_id === attachmentId || enhancedAttachment?.human_id === attachmentId) &&
        enhancedAttachment?.media_type !== '3D' &&
        !!_.find(enhancedAttachment.product_packet_props || [], {
          partner_id: partnerIdInt,
          critical: true,
        }),
    )

  let samplesQuery = `(((parentId: ${productId} OR parentId: ${parentProductId}) AND partnerId: ${partnerId}) AND (evaluationStatus.code: 101 OR evaluationStatus.code: 102))`
  let { content: samples = [] } = await ctx.loaders.search.getSamplesPage.load({
    q: samplesQuery,
  })

  let filteredSamples = samples && samples.filter(sample => sample.workspaceContext === workspaceContext || sample.sampleType.code === 200 || sample.sampleType.code === 135)

  let criticalDiscussionsQuery = {
    q: `((relatedResources:${productId} OR relatedResources: ${parentProductId}) AND (security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (critical:true))`,
  }
  let {
    content: criticalDiscussions = [],
  } = await ctx.loaders.search.searchDiscussionsElastic.load(
    criticalDiscussionsQuery,
  )
  let workspaceContextQuery =
    workspaceContext && workspaceContext !== ''
      ? `AND (workspaceContext: ${workspaceContext}) `
      : ''
  let measurementSetsQuery = {
    q: `((security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (parentId: ${productId}) ${workspaceContextQuery} AND (statusId:200))`,
  }
  let {
    content: measurementSets = [],
  } = await ctx.loaders.search.getMeasurementSets.load(measurementSetsQuery)
  let claimsQuery = {
    q: `((security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (parentId: ${productId}) ${workspaceContextQuery} AND (statusId:501))`,
  }
  let { content: claims = [] } = await ctx.loaders.search.getClaimsElastic.load(
    claimsQuery,
  )
  let bomsQuery = {
    q: `((security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (parentId: ${productId}) ${workspaceContextQuery} AND (statusId:501))`,
  }
  let { content: boms = [] } = await ctx.loaders.search.getBomElastic.load(
    bomsQuery,
  )
  let constructionsQuery = {
    q: `((security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (parentId: ${productId} ${workspaceContextQuery} AND archived:false))`,
  }
  let {
    content: constructions = [],
  } = await ctx.loaders.search.getProductDetailsElastic.load(
    constructionsQuery,
  )
  let watchlistQuery = {
    q: `((security.merchVendors: ${partnerId} OR security.bps: ${partnerId}) AND (parentId: ${productId}) ${workspaceContextQuery} AND (statusId:501))`,
  }
  let {
    content: watchlists = [],
  } = await ctx.loaders.search.searchWatchlist.load(watchlistQuery)

  const { parentDiscussionIds, criticalDiscussionIds, criticalThreadIds } = criticalDiscussions.reduce(({ parentDiscussionIds, criticalDiscussionIds, criticalThreadIds }, discussions) => {
    if (discussions.discussionDto && discussions.discussionDto.critical === true) {
      criticalDiscussionIds.push(discussions.discussionDto.id)
    }
    if (discussions.discussionThreadDtos && discussions.discussionThreadDtos.length) {
      discussions.discussionThreadDtos.map(thread => {
        if (thread.critical) {
          parentDiscussionIds.push(thread.parentId)
          criticalThreadIds.push(thread.id)
        }
      })
    }
    return {
      parentDiscussionIds: parentDiscussionIds.filter((id, index) => parentDiscussionIds.indexOf(id) === index),
      criticalDiscussionIds: criticalDiscussionIds.filter((id, index) => criticalDiscussionIds.indexOf(id) === index),
      criticalThreadIds: criticalThreadIds.filter((id, index) => criticalThreadIds.indexOf(id) === index),
    }
  }, { parentDiscussionIds: [], criticalDiscussionIds: [], criticalThreadIds: [] })

  const allCriticalIds = [...criticalDiscussionIds, ...criticalThreadIds]
  let criticalDiscussionAttachments
  let criticalDiscussionAttachmentsIds = []
  if (allCriticalIds.length > 0) {
    criticalDiscussionAttachments = await ctx.loaders.search.searchAttachmentsByParentResources.load(allCriticalIds)
    criticalDiscussionAttachmentsIds = (criticalDiscussionAttachments.content || [])
      .filter(attachment => attachment?.media_type !== '3D')
      .map(attachment => attachment.id)
  }

  const filteredProductPacketDiscussionAttachments = discussionAttachments.filter(isProductPacketAttachments)
  outputMap = {
    productId: productId,
    partnerId: partnerId,
    workspaceContext: workspaceContext,
    productAttachments: productAttachments.filter(isProductPacketAttachments),
    discussionAttachments: _.union(criticalDiscussionAttachmentsIds, filteredProductPacketDiscussionAttachments),
    discussions: _.union(parentDiscussionIds, criticalDiscussionIds),
    sample: filteredSamples.map(sample => sample.id) || [],
    measurementSets: measurementSets.map(measurement => measurement.id) || [],
    claims: claims.map(claim => claim.id) || [],
    productBoms: boms.filter(bom => bom.type === 1).map(bom => bom.id) || [],
    packagingBoms: boms.filter(bom => bom.type  === 2).map(bom => bom.id) || [],
    constructions: constructions.map(construction => construction.id) || [],
    watchlists: watchlists.map(watchlist => watchlist.id) || [],
  }
  return outputMap
}

        product = await ctx.loaders.product.updateProduct.load(input)
      }
      const copyProductRedirect = await copyProductToProduct(copyProduct, ctx)
      if (copyProductRedirect) {
        product.copyId = copyProductRedirect.copyId
        product.copyProductRequest = copyProductRedirect.requestId
        product.copyProductResources = copyProductRedirect.resources

        input &&
        input.removedProductTemplateAttachments &&
        input.removedProductTemplateAttachments.length
      ) {

        const v3Attachments = []
        input.removedProductTemplateAttachments.forEach(attachment => {
          if (attachment.humanId) {
            v3Attachments.push(attachment)
          } else if (attachment.documentId) {
            v2Attachments.push(attachment)
          }

            v2Attachments.map(attachment => attachment.documentId),
            ctx,

            v3Attachments.map(attachment => attachment.humanId),
            ctx,
          )
          await ctx.loaders.attachment
            .archiveAttachmentBulkV3(permissionJWT)
            .load({ attachments: v3Attachments })
        }
      }
      return product

        )
      }
      return response
