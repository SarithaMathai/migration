export const checkForFacility = async (specification) => {
  const impressions = specification.impressions
  let dyeingAndFinishingFacilities = []
  if (impressions && impressions.length > 0) {
    impressions.forEach(async impression => {
      if (impression.facilities && impression.facilities.length > 0) {
        impression.facilities.forEach(async fac => {
          if (_.get(fac, 'type.code', null) === 877) {
            if (fac.facilityId) {
              dyeingAndFinishingFacilities.push(fac.facilityId)
            }
          }
        })
      }
    })
  }
  return dyeingAndFinishingFacilities.length > 0
}

export const bulkAssociateColorwayFilesWithTeams = async (sparkFabricInput, bulkAttachmentIds, createdFabric, ctx, associateAttachmentTeamDtoT) => {
  const associateAttachmentTeamDto = _.cloneDeep(associateAttachmentTeamDtoT)

  let bulkAssociateAttachmentDto = []
  bulkAssociateAttachmentDto = createdFabric.colorways.map(colorway => {
    const inputColorway = _.find(sparkFabricInput.colorways, (item) => {
      return item.name === colorway.name || item.thumbnailId === colorway.thumbnailId
    })

    if (!inputColorway) {
      return null
    }

    bulkAttachmentIds.push(...inputColorway.files)
    return {
      teamsToUpdateDto: { teamsToAdd: inputColorway.teamsToAdd },
      parentId: colorway.humanId,
      relatedResourceIds: [createdFabric.humanId],
      humanIds: inputColorway.files,
    }
  }).filter(it => !!it)

  if (bulkAttachmentIds.length > 0) {
    const attachmentsPermissionJWT = await getUserPermissionsJWT(bulkAttachmentIds, ctx)

    let associatedBulkList = bulkAssociateAttachmentDto
    if (associateAttachmentTeamDto) {
      bulkAssociateAttachmentDto.push(associateAttachmentTeamDto)
    }
    const bulkAttachmentWrapper = {
      humanIds: bulkAttachmentIds,
      associatedBulkList,
    }
    await ctx.loaders.attachment.bulkAssociateAttachmentTeams(attachmentsPermissionJWT).load(bulkAttachmentWrapper)
  }
}

export const getFabricReportCSV = async (specification, ctx) => {
  const impressions = specification && specification.impressions
  let dyeingAndFinishingFacilities = []
  if (impressions && impressions.length > 0) {
    impressions.forEach(async impression => {
      if (impression.facilities && impression.facilities.length > 0) {
        impression.facilities.forEach(async fac => {
          if (_.get(fac, 'type.code', null) === 877) {
            if (fac.facilityId) {
              dyeingAndFinishingFacilities.push(fac.facilityId)
            }
          }
        })
      }
    })
  }
  const resolvedFacilities = []
  if (dyeingAndFinishingFacilities.length > 0) {
    await Promise.all(
      dyeingAndFinishingFacilities.map(async facilityId => {
        await ctx.loaders.location.getLocationById(facilityId).load()
          .then(res => {
            logger.info(`Completed Data Fetch for facilityId: ${facilityId}`, ctx.logContext)
            resolvedFacilities.push(res)
          })
      }),
    )
  }

  if (resolvedFacilities.length > 0) {
    const resolvedFacility = resolvedFacilities.map(fac => fac.countryOfOrigin).join(', ') || ''
    const isGVSorInternal = specification.gvs ? 'Gvs' : 'Internal'
    logger.info(`Completed Data Fetch for specification.parentHumanId: ${specification.parentHumanId} ,${specification.humanId},${isGVSorInternal},${resolvedFacility}`, ctx.logContext)

    return `${specification.parentHumanId},${specification.humanId},${isGVSorInternal},${resolvedFacility}`
  } else {
    return ''
  }
}

export const searchClaimsForParentResource = async (parentId, ctx) => {
  const permissionJWT = await getUserPermissionsJWT(parentId, ctx)
  const pagedClaims = await ctx.loaders.search.searchClaimsByProxyIds(permissionJWT).load({ q: `parentId:"${parentId}"`, proxyIds: [parentId], page: 0, size: 100 })
  return pagedClaims.content
}

const removeAttachments = async ({ fileHumanIds, ctx }) => {
  const permission = await getUserPermissionsJWT(fileHumanIds, ctx)
  const archiveAttachmentPromises = fileHumanIds?.map(it => ctx.loaders.attachment.archiveAttachmentV3(permission).load(it))
  await Promise.all(archiveAttachmentPromises)
}

export const buildUpdatedColorwaysForNewAttachments = (colorways, attachments, copiedIdToSourceParentMap) => {
  if (isEmptyList(colorways) || isEmptyList(attachments)) {
    return colorways || []
  }

  if (!copiedIdToSourceParentMap || typeof copiedIdToSourceParentMap !== 'object') {
    return colorways || []
  }


  const clonedColorways = _.cloneDeep(colorways)

  return clonedColorways.map(colorway => {
    const matchingAttachmentIds = attachments
      .filter(attachment => {
        const sourceParentId = copiedIdToSourceParentMap[attachment.humanId]
        return sourceParentId === colorway.humanId
      })
      .map(attachment => attachment.humanId)
      .filter(it => !!it)

    return {
      ...colorway,
      files: matchingAttachmentIds,
      thumbnailId: matchingAttachmentIds[0],
    }
  })
}

export const bulkAssociateMaterialFilesAndColorwayFiles = async ({ inputMaterial, createdMaterial, ctx }) => {
  if (materialHasFiles(inputMaterial)) {
    const associateAttachmentTeamDto = {
      teamsToUpdateDto: { teamsToAdd: inputMaterial.teamsToAdd },
      parentId: createdMaterial.humanId,
      humanIds: inputMaterial.files,
      relatedResourceIds: [],
    }
    if (materialHasColorways(inputMaterial)) {
      const bulkAttachmentIds = inputMaterial.files || []
      await bulkAssociateColorwayFilesWithTeams(inputMaterial, bulkAttachmentIds, createdMaterial, ctx, associateAttachmentTeamDto)
    } else {
      const attachmentPermissionJWT = await getUserPermissionsJWT(inputMaterial.files, ctx)
      await ctx.loaders.attachment.associateAttachmentTeams(attachmentPermissionJWT).load(associateAttachmentTeamDto, { humanId: createdMaterial.humanId })
    }
  } else if (materialHasColorways(inputMaterial)) {
    const bulkAttachmentIds = []
    await bulkAssociateColorwayFilesWithTeams(inputMaterial, bulkAttachmentIds, createdMaterial, ctx, null)
  }
}

      return ctx.loaders.fabric.getByID(permissionJWT).load(spec.parentHumanId)
    },
    claims: (spec, args, ctx) => searchClaimsForParentResource(spec.humanId, ctx),
    colorSamples: async (spec, args, ctx) => {
      const relationship = await ctx.loaders.relationship.getWhereUsedByID.load({
        id: spec.humanId,
        type: 'sample',
      })

      if (relationship?.nodes?.length) {
        const totalCount = relationship.nodes.length
        const paging = {
          page: 0,
          totalElement: totalCount,
          last: totalCount <= CONNECTIONS_PER_PAGE,
          totalPages: Math.ceil(totalCount / CONNECTIONS_PER_PAGE),
          size: CONNECTIONS_PER_PAGE,
        }
        const sampleIds = relationship?.nodes?.map(node => node.id).slice(0, CONNECTIONS_PER_PAGE)

        const permissionJWT = await getUserPermissionsJWT(sampleIds, ctx)
        const samples = await ctx.loaders.sampleV2.getSamplesByIdsV2(permissionJWT).load(sampleIds)
        const colorSamples = samples.filter(sample => isColorId(sample.assetHumanId))

        return {
          content: colorSamples,
          paging: paging,
        }
      }
      return {
        content: [],
        paging: NO_RESULTS,
      }
    },
    artworkSamples: async (spec, args, ctx) => {
      const relationship = await ctx.loaders.relationship.getWhereUsedByID.load({
        id: spec.humanId,
        type: 'sample',
      })

      if (relationship.nodes.length) {
        const totalCount = relationship.nodes.length
        const paging = {
          page: 0,
          totalElement: totalCount,
          last: totalCount <= CONNECTIONS_PER_PAGE,
          totalPages: Math.ceil(totalCount / CONNECTIONS_PER_PAGE),
          size: CONNECTIONS_PER_PAGE,
        }
        const sampleIds = relationship.nodes.map(node => node.id).slice(0, CONNECTIONS_PER_PAGE)

        const permissionJWT = await getUserPermissionsJWT(sampleIds, ctx)
        const samples = await ctx.loaders.sampleV2.getSamplesByIdsV2(permissionJWT).load(sampleIds)
        const artworkSamples = samples.filter(sample => sample.assetHumanId && sample.assetHumanId.startsWith(RESOURCE_TYPES_PREFIXES.Artworks))

        return {
          paging: paging,
          content: artworkSamples,
        }
      }
        return {
          content: [],
          paging: NO_RESULTS,
      }
    },
    evaluation: async (spec) => {
      if (spec && spec.evaluation) {
        return spec.evaluation
      } else {
        return null
      }
    },
    assignees: (spec, args, ctx) => spec?.assignees?.map(assignee => ctx.loaders.userAttributes.getUserByID.load(assignee)) || [],
  },
