import cache from '../utils/cache'
import loadListing from '../utils/loadListing'
import { deepToCamelCase } from '../utils/convertFunctions'
import { SPARK_GATEWAY_API_KEY } from '../config/constants'

import SparkService from './SparkService'

export default class LdapDirectoryService extends SparkService {
  constructor (endpoint, token, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.ldapKey = SPARK_GATEWAY_API_KEY

    this.headers = {
      Authorization: `Bearer ${token}`,
      Accept: 'application/json',
    }

    this.getImage = cache(
      'LDAP_Directory',
      loadListing(
        id =>
          `${this.endpoint}/users?key=${
            this.ldapKey
          }&id=${id}`,
        this.headers,
        {
          transform: directory => deepToCamelCase(directory),
        },
        this.logContext,
      ),
    )
  }
}
