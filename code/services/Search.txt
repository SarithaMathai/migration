        )}`,
      this.headers,
      {
        transform: response => {
          let transformedProduct =
            response &&
            response.productsPaged &&
            response.productsPaged.content &&
            response.productsPaged.content.map(product => product.human_id)
          transformedProduct = transformedProduct || []
          let receivedNotEvaluatedByWorkspaceCount =
            response && response.receivedNotEvaluatedByWorkspaceCount
              ? response.receivedNotEvaluatedByWorkspaceCount
              : 0
          let workspaceProductDashboard = response && response.workspaceProductDashboard
          return {
            products: transformedProduct,
            receivedNotEvaluatedByWorkspaceCount: receivedNotEvaluatedByWorkspaceCount,
            workspaceProductDashboard: workspaceProductDashboard,
          }
        },
      },
      this.logContext,
    )
    this.exportFabricCosting = loadListing(
      params => {
        const formattedParams = params

        if (params.nestedQueryList.length > 0) {
          params.nestedQueryList.forEach((item, index) => {
            formattedParams[`nestedQueriesList[${index}].path`] = item.path
            formattedParams[`nestedQueriesList[${index}].query`] = item.query
            formattedParams[`nestedQueriesList[${index}].type`] = item.type
          })
        }
        delete formattedParams.nestedQueryList

        return `${this.endpoint}/materials/v1/export/costing?${qs.stringify(
          formattedParams,
        )}`
      },
      this.headers,
      this.logContext,
    )

    this.exportComboICSReport = loadListing(
      `${this.endpoint}/combinations/v1/comboICSExport`,
      this.headers,
      this.logContext,
    )

    this.exportMaterials = postOne(
      `${this.endpoint}/materials/v2/export`,
      this.headers,
      {
        transform: body => deepToCamelCase(body),
        transformRequest: body => deepToSnakeCase(body),
      },
      this.logContext,
    )

    this.searchMaterialsV2 = postOne(
      `${this.endpoint}/materials/v2/search`,
      this.headers,
      {
        transform: body => deepToCamelCase(body),
        transformRequest: body => deepToSnakeCase(body),
      },
      this.logContext,
    )
    this.multiRequestMaterialSearch = postOne(
      `${this.endpoint}/materials/v2/multiSearch`,
      this.headers,
      {
        transform: body => deepToCamelCase(body),
        transformRequest: body => deepToSnakeCase(body),
      },
      this.logContext,
    )
    this.searchAttachmentsByParentAndRelatedResource = loadOne(
      (queryParams) =>
        `${this.endpoint}/attachments/v1?parentIds=${queryParams[0].parentResourceId}&relatedIds=${queryParams[0].relatedResourceId}&size=10000`,
      this.headers,
      {
        maxBatchSize: 75,
        transform: (ids, json) =>
          deepToCamelCase(
            ids.map(({parentResourceId}) =>
              json.content.filter(attachment =>
                attachment.parent_resource === parentResourceId,
              ),
            ),
          ),
      },
      this.logContext,
    )
    this.searchAttachmentsByParentResources = loadListing(
      parentIds =>
        `${this.endpoint}/attachments/v1?parentIds=${parentIds.join(
          ',',
        )}&size=10000`,
      this.headers,
      {
        transform: results => results && deepToCamelCase(results),
      },
      this.logContext,
    )
    this.searchRfidLocations = loadListing(
      queryParams => `${this.endpoint}/rfid_tags/v1?${qs.stringify(queryParams)}`,
      this.headers,
      {
        transform: results => results && deepToCamelCase(results),
      },
      this.logContext,
    )
    this.searchLatestRfidLocations = loadListing(
      queryParams => `${this.endpoint}/rfid_tags/v1/latest-by-tag?${qs.stringify(queryParams)}`,
      this.headers,
      {
        transform: results => results && deepToCamelCase(results),
      },
      this.logContext,
    )
    this.getRfidSuggestions = loadListing(
      `${this.endpoint}/rfid_tags/v1/suggest`,
      this.headers,
      {},
      this.logContext,
    )
  }
