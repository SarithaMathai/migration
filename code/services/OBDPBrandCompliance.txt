import { deepToCamelCase } from '../utils/convertFunctions'
import loadOne from '../utils/loadOne'
import { SPARK_GATEWAY_API_KEY } from '../config/constants'
import SparkService from './SparkService'
import loadListing from '../utils/loadListing'

export default class BrandComplianceService extends SparkService {
  constructor (endpoint, token, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.apiKey = SPARK_GATEWAY_API_KEY
    this.headers = {
      Authorization: `Bearer ${token}`,
      Accept: 'application/json',
    }

    this.getBrandComplianceProductRecordById = loadOne(id =>
      `${this.endpoint}/product_records/${id}?key=${this.apiKey}`,
    this.headers,
    {
      transform: (id, response) => [deepToCamelCase(response)],
      batch: false,
      logContext,
    },
    )

    this.getBrandComplianceProductRecordByIds = ids => {
      return loadListing(
        `${this.endpoint}/product_records?product_ids=${ids}&key=${this.apiKey}`,
        this.headers,
        {
          transform: response => deepToCamelCase(response),
        }, logContext,
      ).load()
    }
  }
}
