import { deepToCamelCase, deepToSnakeCase } from '../utils/convertFunctions'
import { find } from 'lodash'
import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import loadOne from '../utils/loadOne'
import postOne from '../utils/postOne'
import putOne from '../utils/putOne'

import SparkService from './SparkService'

export default class WorkspaceServiceV2 extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.headers = headers
    this.endpoint = endpoint

    this.getWorkspaceByIdV2 = loadOne(
      id => `${this.endpoint}?ids=${id}`,
      { ...this.headers },
      {
        transform: (ids, json) =>
          deepToCamelCase(ids.map(workspace_human_id => find(json, { workspace_human_id }) || null)),
        fetchAclToken: true,
      },
      logContext,
    )

    this.getWorkspacesByIdsV2 = permissionJWT =>
      loadListing(
        endpoint,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transform: workspaces => deepToCamelCase(workspaces),
        },
        logContext,
      )

    // MUTATIONS
    this.createWorkspaceV2 = postOne(endpoint,
      this.headers,
      {
        transform: workspace => deepToCamelCase(workspace),
        transformRequest: workspace => deepToSnakeCase(workspace),
      },
      this.logContext,
    )

    this.updateWorkspaceV2 = (permissionJWT, validateUnique) =>
      putOne(
        ({ id }) => `${this.endpoint}/${id}?validateUnique=${Boolean(validateUnique)}`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transformRequest: workspace => deepToSnakeCase(workspace),
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.changeWorkspace = permissionJWT => putOne(
      ({ newWorkspaceId }) => `${this.endpoint}/${newWorkspaceId}/change_resource`,
      { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
      {
        transform: data => data && deepToCamelCase(data),
        transformRequest: data => data && deepToSnakeCase(data),
        getDataLoader: () => this.getByID,
        primeKey: workspace => workspace.workspaceId,
      },
      this.logContext,
    )

    this.addResourcesToWorkspaceV2 = permissionJWT =>
      postOne(
        ({ workspaceId }) => `${endpoint}/${workspaceId}/resources/bulk`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.removeWorkspaceResourcesV2 = permissionJWT =>
      deleteOne(
        ({ workspaceId, resourceType, resourceIds }) =>
          `${endpoint}/${workspaceId}/resources/delete/bulk?resourceType=${resourceType}&resourceIds=${resourceIds}`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          getDataLoader: () => this.getByID,
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.addTeamsToWorkspaceV3 = permissionJWT =>
      postOne(
        ({ workspaceId }) => `${endpoint}/${workspaceId}/teams/bulk`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          getDataLoader: () => this.getByID,
          transformRequest: teamBody => deepToSnakeCase(teamBody),
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.removeTeamsFromWorkspaceV3 = permissionJWT =>
      postOne(
        ({ workspaceId }) => `${endpoint}/${workspaceId}/teams/remove/bulk`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          getDataLoader: () => this.getByID,
          transformRequest: teamBody => deepToSnakeCase(teamBody),
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.removeWorkspaceBusinessPartnerV2 = permissionJWT =>
      deleteOne(
        ({ workspaceId, partnerId }) => `${endpoint}/${workspaceId}/partner/${partnerId}`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          getDataLoader: () => this.getByID,
          transformRequest: teamBody => deepToSnakeCase(teamBody),
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.dropWorkspaceBusinessPartnerV2 = resourceBPPermissionJWT =>
      postOne(
        ({ workspaceId, partnerId }) =>
          `${endpoint}/${workspaceId}/drop-partner/${partnerId}`,
        { ...this.headers, 'SPARK-Capability-Token': resourceBPPermissionJWT },
        {
          getDataLoader: () => this.getByID,
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )

    this.unDropWorkspaceBusinessPartnerV2 = resourcePermissionJWT =>
      postOne(
        ({ workspaceId, partnerId }) =>
          `${endpoint}/${workspaceId}/undrop-partner/${partnerId}`,
        { ...this.headers, 'SPARK-Capability-Token': resourcePermissionJWT },
        {
          getDataLoader: () => this.getByID,
          transform: workspace => deepToCamelCase(workspace),
        },
        this.logContext,
      )
    this.dropUndropWorkspaceBusinessPartnerV2 = permissionJWT => postOne(
      ({ workspaceId }) =>
        `${this.endpoint}/${workspaceId}/drop-undrop-partner`, {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      }, {
        getDataLoader: () => this.getByID,
        transform: product => deepToCamelCase(product),
        transformRequest: product => deepToSnakeCase(product),
      },
    )
  }
}
