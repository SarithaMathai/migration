import qs from 'querystring'

import { deepToCamelCase } from '../utils/convertFunctions'
import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import postOne from '../utils/postOne'

import SparkService from './SparkService'

export default class RecentlyViewedService extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = headers

    this.getPage = loadListing(
      params => `${this.endpoint}?${qs.stringify(params)}`,
      this.headers,
      {},
      this.logContext,
    )

    this.deleteRecentlyViewedByPartner = deleteOne(
      ({ resourceType, partnerId, resourceIds }) =>
        `${
          this.endpoint
        }/${resourceType}/partner/${partnerId}?resourceIds=${resourceIds}`,
      this.headers,
      {},
      this.logContext,
    )
    this.addRecentlyViewed = postOne(
      `${this.endpoint}`,
      this.headers,
      {
        transform: data => data && deepToCamelCase(data),
      },
      this.logContext,
    )
  }
}
