import { deepToCamelCase, deepToSnakeCase, buildCommaSeparatedString } from '../utils/convertFunctions'
import { find } from 'lodash'
import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import loadOne from '../utils/loadOne'
import postOne from '../utils/postOne'
import putOne from '../utils/putOne'

import SparkService from './SparkService'
export default class DiscussionService extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = headers
    this.endpointv1 = endpoint + '/enterprise_product_development_discussions/v1'
    this.endpointv2 = endpoint + '/enterprise_product_development_discussions/v2'

    this.getDiscussionsCount = loadListing(
      ({ resourceId, resourceType }) =>
        `${
          this.endpointv1
        }/discussions/count?resourceType=${resourceType}&resourceId=${resourceId}`,
      this.headers,
      {
        // transform: discussion => discussion,
        // batch: false
      },
      this.logContext,
    )

    this.getUnsentDiscussions = loadListing(
      () => `${this.endpoint}/api/admin/unsent`,
      this.headers,
      {
      },
      this.logContext,
    )

    this.getByIDV2 = loadOne(
      id => `${this.endpointv2}?ids=${id}&page=1&size=${id.length || 20}`,
      this.headers,
      {
        transform: (ids, json) =>
          deepToCamelCase(ids.map(discussion_id => find(json, { discussion_id }) || null)),
        fetchAclToken: true,
      },
      this.logContext,
    )

    this.getByIDsV2 = permissionJWT => loadListing(
      ids => `${this.endpointv2}?ids=${buildCommaSeparatedString(ids)}&page=1&size=${ids.length || 20}`,
      { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
      {
        transform: discussions => deepToCamelCase(discussions),
      },
      this.logContext,
    )

    // createSampleDiscussion API will move to v2 controller after the cleanup
    this.createSampleDiscussion = postOne(
      `${this.endpointv1}/discussion_message`,
      this.headers,
      {
        transformRequest: sparkSampleDiscussion =>
          deepToSnakeCase(sparkSampleDiscussion),
        transform: discussion => discussion && deepToCamelCase(discussion),
        primeKey: discussion => discussion.discussionId,
      },
      this.logContext,
    )

    this.bulkCreateSampleDiscussions = postOne(
      `${this.endpointv1}/discussion_message/bulk`,
      this.headers,
      {
        transformRequest: sparkSampleDiscussion =>
          deepToSnakeCase(sparkSampleDiscussion),
        transform: discussion => discussion && deepToCamelCase(discussion),
      },
      this.logContext,
    )

    this.createV2 = postOne(
      `${this.endpointv2}/discussion`,
      this.headers,
      {
        transformRequest: sparkDiscussion => deepToSnakeCase(sparkDiscussion),
        transform: discussion => discussion && deepToCamelCase(discussion),
        primeKey: discussion => discussion.discussionId,
      },
      this.logContext,
    )

    this.getDiscussionsV2 = permissionJWT =>
      loadListing(
        ({ ids }) => `${this.endpointv2}?ids=${ids}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
          // batch: false
        },
        this.logContext,
      )
    // Get all discussion is a post call now. due to the length of the query params get call is failing some places
    this.getDiscussionsByPost = permissionJWT =>
      postOne(
        () => `${this.endpointv2}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
          // batch: false
        },
        this.logContext,
      )

    this.getThread = loadOne(
      (id) => `${this.endpointv2}/threads?ids=${id}`,
      this.headers,
      {
        transform: (ids, json) => {
          return deepToCamelCase(ids.map(discussion_thread_id => find(json, { discussion_thread_id }) || null))
        },
        fetchAclToken: true,
      },
      this.logContext,
    )

    // Get discussion threads is a post call now. due to the length of the query params get call is failing some places
    this.getThreadsByPost = permissionJWT =>
      postOne(
        () => `${this.endpointv2}/threads`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
          // batch: false
        },
        this.logContext,
      )

    this.updateV2 = permissionJWT =>
      putOne(
        sparkDiscussion =>
          `${this.endpointv2}/discussion/${sparkDiscussion.discussionId}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
          primeKey: discussion => discussion.discussionId,
        },
        this.logContext,
      )
    this.updateParticipantsV2 = permissionJWT =>
      postOne(
        ({ id }) => `${this.endpointv2}/discussion/${id}/participants`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: participants => deepToSnakeCase(participants),
          transform: discussion => discussion && deepToCamelCase(discussion),
          primeKey: discussion => discussion.discussionId,
        },
        this.logContext,
      )
    this.deleteParticipantsV2 = permissionJWT =>
      postOne(
        ({ id }) => `${this.endpointv2}/discussion/${id}/participants/delete`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: participants => deepToSnakeCase(participants),
          transform: (id, discussion) =>
            discussion && deepToCamelCase(discussion),
          batch: false,
        },
        this.logContext,
      )
    this.deleteV2 = permissionJWT =>
      deleteOne(
        id => `${this.endpointv2}/discussion/${id}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
        },
        this.logContext,
      )
    this.createReplyV2 = permissionJWT =>
      postOne(
        discussionId => `${this.endpointv2}/discussion/${discussionId}/thread`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: reply => deepToSnakeCase(reply),
          getDataLoader: reply => reply,
          transform: reply => reply && deepToCamelCase(reply),
          primeKey: reply => reply.id,
        },
        this.logContext,
      )
    this.dropPartnerFromDiscussionIds = permissionJWT =>
      postOne(
        `${this.endpointv2}/drop-partner`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
        },
        this.logContext,
      )
    this.unDropPartnerFromDiscussionIds = permissionJWT =>
      postOne(
        `${this.endpointv2}/undrop-partner`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
        },
        this.logContext,
      )
    this.discussionReadByUsers = permissionJWT =>
      postOne(
        `${this.endpointv2}/discussion/read-by-users`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: discussion => discussion && deepToCamelCase(discussion),
        },
        this.logContext,
      )
    this.updateReplyV2 = permissionJWT =>
      putOne(
        sparkDiscussionReply =>
          `${this.endpointv2}/thread/${sparkDiscussionReply.id}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: sparkDiscussionReply =>
            deepToSnakeCase(sparkDiscussionReply),
          getDataLoader: () => reply => reply,
          transform: reply => reply && deepToCamelCase(reply),
          primeKey: reply => reply.id,
        },
        this.logContext,
      )
    this.deleteReplyV2 = permissionJWT =>
      deleteOne(
        ({ replyId }) => `${this.endpointv2}/thread/${replyId}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {},
        this.logContext,
      )
    this.updateToCritical = permissionJWT =>
      putOne(
        criticalObject => criticalObject.discussionThreadId ? `${this.endpointv2}/discussion/${criticalObject.discussionId}/thread/${criticalObject.discussionThreadId}/details` : `${this.endpointv2}/discussion/${criticalObject.discussionId}/details`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: reply => reply && deepToCamelCase(reply),
        },
        this.logContext,
      )
    this.updateDiscussionEditable = permissionJWT =>
      putOne(
        discussionEditableObject => discussionEditableObject.discussionThreadId ? `${this.endpointv2}/discussion/${discussionEditableObject.discussionId}/thread/${discussionEditableObject.discussionThreadId}/details` : `${this.endpointv2}/discussion/${discussionEditableObject.discussionId}/details`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: editUpdateObject => editUpdateObject && deepToCamelCase(editUpdateObject),
        },
        this.logContext,
      )
    this.updateTag = permissionJWT =>
      putOne(
        tagUpdateObject => `${this.endpointv2}/discussion/${tagUpdateObject.discussionId}/details`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: tagUpdateObject => tagUpdateObject && deepToCamelCase(tagUpdateObject),
        },
        this.logContext,
      )
  }
}
