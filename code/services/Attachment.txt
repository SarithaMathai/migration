import qs from 'querystring'
import { find } from 'lodash'

import { deepToCamelCase, deepToSnakeCase } from '../utils/convertFunctions'
import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import loadOne from '../utils/loadOne'
import postOne from '../utils/postOne'
import putOne from '../utils/putOne'
import SparkService from './SparkService'

export default class AttachmentService extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = headers

    this.getAttachmentsV3 = permissionJWT =>
      loadListing(
        humanIds => `${this.endpoint}/attachments/v3?humanIds=${humanIds}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {},
        this.logContext,
      )

    this.getAttachmentsByPostV3 = permissionJWT =>
      postOne(
        () => `${this.endpoint}/attachments/v3/attachmentids`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {

        },
        this.logContext,
      )

    this.getAttachmentsByIdsAndAuthorByPostV3 = () => postOne(`${this.endpoint}/attachments/v3/ids`,
      {
        ...this.headers,
      },
      {
        transformRequest: obj => deepToSnakeCase(obj),
      },
      this.logContext,
    )

    this.archiveAttachmentV3 = permissionJWT =>
      deleteOne(
        humanId => `${this.endpoint}/attachments/v3/${humanId}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {},
        this.logContext,
      )

    this.getPreSignedURLs = permissionJWT => loadListing(
      ids => {
        const requestParamString = qs.stringify({ ids })
        const url = `${this.endpoint}/attachments/v3/pre-signed-urls?${requestParamString}`
        return url
      },
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {},
      this.logContext,
    )

    this.archiveAttachmentBulkV3 = permissionJWT =>
      postOne(
        () => `${this.endpoint}/attachments/v3/bulk-delete`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: attachments => deepToSnakeCase(attachments),
        },
        this.logContext,
      )

    this.deleteAttachmentV3 = permissionJWT =>
      deleteOne(
        humanId => `${this.endpoint}/attachments/v3/${humanId}/file`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        { skipJsonParse: true },
        this.logContext,
      )

    this.copyAttachmentsV3 = permissionJWT =>
      postOne(
        () => `${this.endpoint}/attachments/v3/copy`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: obj => deepToCamelCase(obj),
          transformRequest: obj => deepToSnakeCase(obj),
          batch: false,
        },
        this.logContext,
      )

    this.associateResourcesV3 = permissionJWT =>
      putOne(
        () => `${this.endpoint}/attachments/v3/add-association`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: obj => deepToCamelCase(obj),
          transformRequest: obj => deepToSnakeCase(obj),
          batch: false,
        },
        this.logContext,
      )

    this.removeResourcesV3 = permissionJWT =>
      putOne(
        () => `${this.endpoint}/attachments/v3/remove-association`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: obj => deepToCamelCase(obj),
          transformRequest: obj => deepToSnakeCase(obj),
          batch: false,
        },
        this.logContext,
      )

    // Get attachments for discussions/discussion threads is a post call now. due to the length of the query params get call is failing some places
    this.getAttachmentsByPost = permissionJWT =>
      postOne(
        () => `${this.endpoint}/attachments/v3/attachmentids`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {},
        this.logContext,
      )

    this.archiveAttachmentBulkV2 = permissionJWT =>
      postOne(
        () => `${this.endpoint}/attachments/v3/bulk-delete`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transformRequest: attachments => deepToSnakeCase(attachments),
        },
        this.logContext,
      )

    this.create = postOne(
      () => `${this.endpoint}/bulk/attachments`,
      this.headers,
      {},
      this.logContext,
    )

    this.search = loadListing(
      ({ resourceId, resourceType }) =>
        `${
          this.endpoint
        }/attachment_search?resourceType=${resourceType}&resources=${resourceId}`,
      this.headers,
      {
        getDataLoader: () => this.getByID,
        primeKey: attachment => attachment.document_id,
      },
      this.logContext,
    )

    this.updateAccess = putOne(
      ({ resourceId }) => `${this.endpoint}/access/resource/${resourceId}`,
      this.headers,
      {
        getDataLoader: () => this.getByID,
        transformRequest: access => deepToSnakeCase(access),
        transform: access => deepToCamelCase(access),
      },
      this.logContext,
    )

    this.bulkUpdateResource = (permissionJWT) => postOne(
      () => `${this.endpoint}/attachments/v3/bulk-update`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        getDataLoader: () => this.getByID,
        transformRequest: attachmentList => deepToSnakeCase({ attachments: attachmentList }),
        transform: attachmentList => deepToCamelCase(attachmentList),
      },
      this.logContext,
    )

    this.updateTagsV3 = permissionJWT =>
      putOne(
        ({ attachmentId }) => `${this.endpoint}/attachments/v3/${attachmentId}/tags`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          getDataLoader: () => this.getByID,
          transformRequest: data => data.tags,
        },
        this.logContext,
      )
    this.bulkUpdateTags = permissionJWT => postOne(
      () => `${this.endpoint}/attachments/v3/bulk-update-tags`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: attachmentTags => deepToSnakeCase({ attachmentTags }),
      },
      this.logContext,
    )

    this.updateAttributes = permissionJWT => postOne(
      () => `${this.endpoint}/attachments/v3/update-attr`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: attachments => deepToSnakeCase(attachments),
      },
      this.logContext,
    )

    this.bulkUpdateAttributes = permissionJWT => postOne(
      () => `${this.endpoint}/attachments/v3/bulk-update-attr`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: attachments => deepToSnakeCase(attachments),
      },
      this.logContext,
    )

    this.associateAttachmentTeams = permissionJWT => postOne(
      ({ humanId }) => `${this.endpoint}/attachments/v3/${humanId}/teams`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: associateAttachmentTeamDto => deepToSnakeCase(associateAttachmentTeamDto),
      },
      logContext,
    )
    this.bulkAssociateAttachmentTeams = permissionJWT => postOne(
      `${this.endpoint}/attachments/v3/bulkUpdate/teams`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: associateAttachmentTeamDtos => deepToSnakeCase(associateAttachmentTeamDtos),
      },
      logContext,
    )

    this.bulkUpdateAttachmentsV2 = permissionJWT => postOne(
      () => `${this.endpoint}/attachments/v3/bulk-update-attachment`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: request => deepToSnakeCase(request),
        transform: permissions => deepToCamelCase(permissions),
      },
      logContext,
    )

    this.publishAttachmentToGallery = permissionJWT => postOne(
      (attachmentId) => `${this.endpoint}/attachments/v3/${attachmentId}/publish`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        skipJsonParse: true,
      },
      logContext,
    )

    this.unpublishAttachmentToGallery = permissionJWT => postOne(
      (attachmentId) => `${this.endpoint}/attachments/v3/${attachmentId}/unpublish`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        skipJsonParse: true,
      },
      logContext,
    )

    this.publishAttachmentToGalleryV3 = permissionJWT => postOne(
      (attachmentId) => `${this.endpoint}/attachments/v3/${attachmentId}/publish`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        skipJsonParse: true,
      },
      logContext,
    )

    this.unpublishAttachmentToGalleryV3 = permissionJWT => postOne(
      (attachmentId) => `${this.endpoint}/attachments/v3/${attachmentId}/unpublish`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        skipJsonParse: true,
      },
      logContext,
    )

    this.getRendersForAttachmentIds = betaMode => loadOne(
      (id) => {
        return `${this.endpoint}/attachments/v3/renders?ids=${id}&betaMode=${betaMode}`
      },
      this.headers,
      {
        transform: (ids, json) => deepToCamelCase(ids.map(attachment_id => find(json.render_attachments, { attachment_id })) || null),
        fetchAclToken: true,
        batch: true,
      },
      logContext,
    )

    this.getRendersForAttachmentIdsByPost = (permissionJWT) =>
      postOne(
        `${this.endpoint}/attachments/v3/renders`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: (json) => (json.render_attachments && deepToCamelCase(json.render_attachments)) || [],
          batch: false,
        },
        this.logContext,
      )

    this.getRendersForAttachmentV3Ids = betaMode => loadOne(
      (id) => {
        return `${this.endpoint}/attachments/v3/renders?ids=${id}&betaMode=${betaMode}`
      },
      this.headers,
      {
        transform: (ids, json) => deepToCamelCase(ids.map(attachment_id => find(json.render_attachments, { attachment_id })) || null),
        fetchAclToken: true,
        batch: true,
      },
      logContext,
    )

    // Get new attachment to clone PDTL attachment
    this.cloneAttachmentV3 = permissionJWT =>
      postOne((id) => {
        return `${this.endpoint}/attachments/v3/${id}/clone`
      },
      { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
      {
        transformRequest: data => deepToSnakeCase(data),
        transform: attachmentIds => deepToCamelCase(attachmentIds),
        batch: false,
      },
      this.logContext,
      )
  }
}
