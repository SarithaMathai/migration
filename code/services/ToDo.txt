import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import postOne from '../utils/postOne'

import SparkService from './SparkService'

export default class ToDoService extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = headers

    this.getPage = loadListing(`${this.endpoint}`, this.headers, {}, logContext)

    this.deleteToDoByBusinessPartner = deleteOne(
      ({ resourceType, partnerId, resourceIds }) =>
        `${
          this.endpoint
        }/${resourceType}/partner/${partnerId}?resourceIds=${resourceIds}`,
      this.headers,
      {},
      this.logContext,
    )

    this.deleteToDo = deleteOne(
      toDoId => `${this.endpoint}/${toDoId}`,
      this.headers,
      {},
      this.logContext,
    )

    this.addToDo = postOne(
      `${this.endpoint}`,
      this.headers,
      {},
      this.logContext,
    )
  }
}
