import _ from 'lodash'

import { IG_DEPARTMENT_PATH, SPARK_GATEWAY_API_KEY } from '../config/constants'
import loadListing from '../utils/loadListing'
import loadOne from '../utils/loadOne'

import SparkService from './SparkService'

export default class DivisionService extends SparkService {
  constructor (endpoint, token, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = {
      Authorization: `Bearer ${token}`,
      Accept: 'application/json',
    }

    this.igKey = SPARK_GATEWAY_API_KEY
    this.searchFilter = '&per_page=1000'
    this.getPage = loadListing(
      () =>
        `${this.endpoint}/${IG_DEPARTMENT_PATH}?key=${this.igKey}${this.searchFilter}`,
      this.headers,
      {
        batch: false,
        transform: divisions => {
          return _.uniqBy(divisions, 'division_id')
        },
      },
      this.logContext,
    )

    this.getByID = loadOne(
      id =>
        `${this.endpoint}/${IG_DEPARTMENT_PATH}?key=${this.igKey}&division_id=${id}`,
      this.headers,
      {
        transform: (ids, divisions) => divisions && divisions.slice(0, 1),
        batch: false,
      },
      this.logContext,
    )
  }
}
