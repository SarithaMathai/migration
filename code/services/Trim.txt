import SparkService from './SparkService'
import _ from 'lodash'
import { deepToCamelCase, deepToSnakeCase } from '../utils/convertFunctions'
import loadOne from '../utils/loadOne'
import postOne from '../utils/postOne'

    super(logContext)
    this.endpoint = `${endpoint}`
    this.headers = headers

    this.keyValuesTransform = values => {
      const formattedValues = []
      for (var key in values) {
        if (Object.prototype.hasOwnProperty.call(values, key)) {
          formattedValues.push({
            code: key,
            description: values[key],
          })
        }
      }
      return _.sortBy(formattedValues, ['description'])
    }

    this.getTrimBatch = loadOne(
      id => `${this.endpoint}?humanIds=${id}`,
      this.headers,
      {
        transform: (ids, json) =>
          deepToCamelCase(ids.map(human_id => _.find(json, { human_id }) || null)),
        fetchAclToken: true,
      },
      this.logContext,
    )

    this.getTrim = permissionJWT =>
      loadOne(
        id => `${this.endpoint}/${id}`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: (id, trim) => [deepToCamelCase(trim)],
        },
        this.logContext,
      )

    this.getTrims = permissionJWT =>
      loadListing(
        ids => `${this.endpoint}?humanIds=${ids}`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: trim => deepToCamelCase(trim) },
        logContext,
      )

      `${this.endpoint}/validate_uniqueness`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transform: response => deepToCamelCase(response),
        transformRequest: request => deepToSnakeCase(request),
      },
      this.logContext,
    )

      this.endpoint,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transform: trim => deepToCamelCase(trim),
        transformRequest: trim => deepToSnakeCase(trim),
        throwOnError: true,
      },
      this.logContext,
    )

        'SPARK-Capability-Token': permissionJWT,
