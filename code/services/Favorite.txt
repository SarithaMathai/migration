import deleteOne from '../utils/deleteOne'
import loadListing from '../utils/loadListing'
import postOne from '../utils/postOne'

import SparkService from './SparkService'

export default class FavoriteService extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = endpoint
    this.headers = headers

    this.getPage = loadListing(
      `${this.endpoint}`,
      this.headers,
      {},
      this.logContext,
    )

    this.create = postOne(this.endpoint, this.headers, {}, this.logContext)

    this.delete = deleteOne(
      ({ id, type }) => `${this.endpoint}/resource/${type}/${id}`,
      this.headers,
      {},
      this.logContext,
    )

    this.updateFavoritesOrder = (favoriteIds) => postOne(
      `${this.endpoint}/reorder`,
      this.headers,
      {
        skipJsonParse: true,
      },
      this.logContext,
    ).load(favoriteIds)

    // Deprecated
    this.deleteById = deleteOne(
      id => `${this.endpoint}/${id}`,
      this.headers,
      {},
      this.logContext,
    )

    this.deleteFavoritesByBusinessPartner = deleteOne(
      ({ resourceType, resourceId, partnerId }) =>
        `${this.endpoint}/${resourceType}/${resourceId}/partner/${partnerId}`,
      this.headers,
      {},
      this.logContext,
    )
  }
}
