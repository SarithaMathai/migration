import _ from 'lodash'

import { deepToCamelCase, deepToSnakeCase } from '../utils/convertFunctions'
import loadListing from '../utils/loadListing'
import loadOne from '../utils/loadOne'
import postOne from '../utils/postOne'
import putOne from '../utils/putOne'

import SparkService from './SparkService'

export default class SampleServiceV2 extends SparkService {
  constructor (endpoint, headers, logContext) {
    super(logContext)
    this.endpoint = `${endpoint}/enterprise_product_development_samples/v2`
    this.exportEndpoint = `${endpoint}/enterprise_product_development_samples/export`
    this.headers = headers

    this.codeValuesTransform = values => {
      const formattedValues = []
      _.keys(values).forEach(code => {
        formattedValues.push({ code, description: values[code] })
      })
      return formattedValues
    }

    this.createSamplesV2 = postOne(
      this.endpoint,
      this.headers,
      {
        transform: sample => deepToCamelCase(sample),
        transformRequest: sample => deepToSnakeCase(sample),
      },
      this.logContext,
    )
    this.createSampleRoundV2 = permissionJWT =>
      postOne(
        ({ sampleId }) => `${this.endpoint}/${sampleId}/round`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transform: sample => deepToCamelCase(sample),
          transformRequest: sample => deepToSnakeCase(sample),
        },
        this.logContext,
      )

    this.updateSamplesV2 = permissionJWT =>
      putOne(
        () => this.endpoint,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transformRequest: samples => deepToSnakeCase(samples),
          transform: sample => deepToCamelCase(sample),
        },
        this.logContext,
      )

    this.updateSampleValueAssessmentData = (permissionJWT, sampleId) =>
      putOne(
        () => `${this.endpoint}/${sampleId}/value-assessment`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        {
          transformRequest: requestData => deepToSnakeCase(requestData),
          transform: responseData => deepToCamelCase(responseData),
        },
        this.logContext,
      )

    this.bulkCreateSampleRounds = permissionJWT => postOne(`${this.endpoint}/rounds`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transformRequest: samples => deepToSnakeCase(samples),
        transform: sample => deepToCamelCase(sample),
      },
      this.logContext,
    )

    this.updateSampleWorkspaceAssociation = (permissionJWT, sampleId, workspaceId) =>
      putOne(() =>
        `${this.endpoint}/${sampleId}/workspace/${workspaceId}`,
      {
        ...this.headers,
        'SPARK-Capability-Token': permissionJWT,
      },
      {
        transform: sample => deepToCamelCase(sample),
      },
      this.logContext,
      )

    this.getSampleById = loadOne(
      id => `${this.endpoint}?ids=${id}`,
      this.headers,
      {
        transform: (ids, json) =>
          deepToCamelCase(ids.map(id => _.find(json, { id }) || null)),
        fetchAclToken: true,
      },
      this.logContext,
    )

    this.getSamplesByIdsV2 = permissionJWT =>
      loadListing(
        ids => `${this.endpoint}?ids=${ids}`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: samples => _.orderBy(deepToCamelCase(samples), 'createdAt', 'desc') },
        this.logContext,
      )

    this.getSamplesByIdsV2ByPost = permissionJWT =>
      postOne(
        () => `${this.endpoint}/getSamplesByIdsV2`,
        {
          ...this.headers,
          'SPARK-Capability-Token': permissionJWT,
        },
        {
          transform: samples => _.orderBy(deepToCamelCase(samples), 'createdAt', 'desc'),
        },
        this.logContext,
      )

    this.getSampleMaterialTypesV2 = loadListing(
      () => `${this.endpoint}/samples/typelist?resource_type=material`,
      this.headers,
      {
        transform: types => deepToCamelCase(types).sampleTypes,
      },
      this.logContext,
    )

    this.getSampleProductTypesV2 = loadListing(
      () => `${this.endpoint}/samples/typelist?resource_type=PRODUCT`,
      this.headers,
      {
        transform: types => deepToCamelCase(types).sampleTypes,
      },
      this.logContext,
    )

    this.getSampleTrackingTypesV2 = loadListing(
      () => `${this.endpoint}/samples/tracking/types`,
      this.headers,
      {
        transform: types => this.codeValuesTransform(types),
      },
      this.logContext,
    )

    this.getSampleLateReasonTypesV2 = loadListing(
      () => `${this.endpoint}/samples/latereason/types`,
      this.headers,
      {
        transform: types => this.codeValuesTransform(types),
      },
      this.logContext,
    )

    this.getMaterialSampleEvaluationTypesV2 = loadListing(
      () => `${this.endpoint}/samples/evaluation/types?resource_type=material`,
      this.headers,
      {
        transform: types => this.codeValuesTransform(types),
      },
      this.logContext,
    )

    this.getProductSampleEvaluationTypesV2 = loadListing(
      () => `${this.endpoint}/samples/evaluation/types?resource_type=PRODUCT`,
      this.headers,
      {
        transform: types => this.codeValuesTransform(types),
      },
      this.logContext,
    )

    this.getFabricSampleTypesV2 = loadListing(
      () => `${this.endpoint}/samples/typelist?resource_type=fabric_sample`,
      this.headers,
      {
        transform: types => deepToCamelCase(types).sampleTypes,
      },
      this.logContext,
    )
    this.getSampleRounds = permissionJWT =>
      loadListing(
        humanId => `${this.endpoint}/${humanId}/rounds`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: samples => deepToCamelCase(samples) },
        this.logContext,
      )

    this.getSamplesByRfidTagIds = permissionJWT =>
      loadListing(
        ids => `${this.endpoint}/rfidTagIds?${ids.map(id => `ids=${encodeURIComponent(id)}`).join('&')}`,  
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: data => deepToCamelCase(data) },
        this.logContext,
      )

    this.dropSamples = permissionJWT =>
      postOne(
        () =>
          `${this.endpoint}/drop-samples`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: samples => deepToCamelCase(samples) },
        this.logContext,
      )

    this.unDropSamples = permissionJWT =>
      postOne(
        () =>
          `${this.endpoint}/undrop-samples`,
        { ...this.headers, 'SPARK-Capability-Token': permissionJWT },
        { transform: samples => deepToCamelCase(samples) },
        this.logContext,
      )

    this.getSampleExports = loadListing(
      () => this.exportEndpoint,
      this.headers,
      {
        transform: exports => deepToCamelCase(exports),
      },
      this.logContext,
    )

    this.requestSampleExport = postOne(
      this.exportEndpoint,
      this.headers,
      {
        transform: response => response.request_id,
      },
      this.logContext,
    )
    this.getWashSampleTypesV2 = loadListing(
      () => `${this.endpoint}/samples/typelist?resource_type=wash_sample`,
      this.headers,
      {
        transform: types => deepToCamelCase(types).sampleTypes,
      },
      this.logContext,
    )

    this.getSampleFormats = loadListing(
      (type) => `${this.endpoint}/samples/formats?type=${type || 'PRODUCT'}`,
      this.headers,
      {
        transform: deepToCamelCase,
      },
      this.logContext,
    )

    this.getSampleEvaluationPurposes = loadListing(
      () => `${this.endpoint}/samples/evaluation/purposes`,
      this.headers,
      {
        transform: purposes => this.codeValuesTransform(purposes),
      },
      this.logContext,
    )

    this.getSampleTypeFormatMappings = loadListing(
      () => `${this.endpoint}/samples/type_formats`,
      this.headers,
      {
        transform: mappings => _.keys(mappings).map(key => ({
          sampleType: +key,
          sampleFormat: mappings[key].code,
        })),
      },
      this.logContext,
    )

    this.getSampleTypesV2 = loadListing(
      (resourceTypes) => `${this.endpoint}/samples/typelist?resource_types=${resourceTypes}`,
      this.headers,
      {
        transform: types => deepToCamelCase(types).sampleTypes,
      },
      this.logContext,
    )
    this.getMaterialSampleFormats = loadListing(
      (type) => `${this.endpoint}/material/samples/formats?type=${type}`,
      this.headers,
      {
        transform: deepToCamelCase,
      },
      this.logContext,
    )

    this.getColorSamplePurposesV2 = loadListing(
      () => `${this.endpoint}/samples/color_sample/purposes`,
      this.headers,
      {
        transform: purposes => this.codeValuesTransform(purposes),
      },
      this.logContext,
    )
    this.getColorSamplesByParentId = loadListing(
      (id) => `${this.endpoint}/parent?id=${id}`,
      this.headers,
      { transform: samples => _.orderBy(deepToCamelCase(samples), 'createdAt', 'desc') },
      this.logContext,
    )
  }
}
