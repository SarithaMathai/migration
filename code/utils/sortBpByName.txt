import { get } from 'lodash'

const sortBpByName = (list) => {
  if (list && Array.isArray(list)) {
    const newList = [...list]
    newList.sort((a, b) => {
      const first = get(a, 'bpName', '').toLowerCase()
      const second = get(b, 'bpName', '').toLowerCase()
      return first > second ? 1 : -1
    })
    return newList
  }
}

export default sortBpByName
